90  	@Test
91  	public void headerMapperParserTest() throws Exception {
92  		SimpleWebServiceOutboundGateway gateway =
93  				TestUtils.getPropertyValue(this.context.getBean("withHeaderMapper"),
94  				"handler", SimpleWebServiceOutboundGateway.class);
95  		DefaultSoapHeaderMapper headerMapper = TestUtils.getPropertyValue(gateway, "headerMapper",
96  				DefaultSoapHeaderMapper.class);
97  		assertThat(headerMapper).isNotNull();
98  
99  		AbstractHeaderMapper.HeaderMatcher requestHeaderMatcher = TestUtils.getPropertyValue(headerMapper,
100 				"requestHeaderMatcher", AbstractHeaderMapper.HeaderMatcher.class);
101 		assertThat(requestHeaderMatcher.matchHeader("foo")).isTrue();
102 		assertThat(requestHeaderMatcher.matchHeader("foo123")).isTrue();
103 		assertThat(requestHeaderMatcher.matchHeader("baz")).isTrue();
104 		assertThat(requestHeaderMatcher.matchHeader("123baz123")).isTrue();
105 		assertThat(requestHeaderMatcher.matchHeader("bar")).isFalse();
106 		assertThat(requestHeaderMatcher.matchHeader("bar123")).isFalse();
107 
108 		AbstractHeaderMapper.HeaderMatcher replyHeaderMatcher = TestUtils.getPropertyValue(headerMapper,
109 				"replyHeaderMatcher", AbstractHeaderMapper.HeaderMatcher.class);
110 		assertThat(replyHeaderMatcher.matchHeader("foo")).isFalse();
111 		assertThat(replyHeaderMatcher.matchHeader("foo123")).isFalse();
112 		assertThat(replyHeaderMatcher.matchHeader("baz")).isFalse();
113 		assertThat(replyHeaderMatcher.matchHeader("123baz123")).isFalse();
114 		assertThat(replyHeaderMatcher.matchHeader("bar")).isTrue();
115 		assertThat(replyHeaderMatcher.matchHeader("bar123")).isTrue();
116 	}

202 		AbstractWebServiceOutboundGateway gateway =
203 				TestUtils.getPropertyValue(this.context.getBean(gatewayName), "handler",
204 				AbstractWebServiceOutboundGateway.class);

206 		if (!soap) {
207 			WebServiceTemplate template = TestUtils.getPropertyValue(gateway, "webServiceTemplate", WebServiceTemplate.class);
208 			template.setMessageFactory(new DomPoxMessageFactory());
209 		}
210 
211 		WebServiceMessageSender messageSender = Mockito.mock(WebServiceMessageSender.class);
212 		WebServiceConnection wsConnection = Mockito.mock(WebServiceConnection.class);
213 		Mockito.when(messageSender.createConnection(Mockito.any(URI.class))).thenReturn(wsConnection);
214 		Mockito.when(messageSender.supports(Mockito.any(URI.class))).thenReturn(true);
215 
216 		Mockito.doAnswer(invocation -> {
217 
218 			Object[] args = invocation.getArguments();
219 			WebServiceMessage wsMessage = (WebServiceMessage) args[0];
220 //				try { // uncomment if you want to see a pretty-print of SOAP message
221 //					Transformer transformer = TransformerFactory.newInstance().newTransformer();
222 //					transformer.setOutputProperty(OutputKeys.INDENT, "yes");
223 //					transformer.transform(new DOMSource(soapMessage.getDocument()), new StreamResult(System.out));
224 //				}
225 //				catch (Exception e) {
226 //					// ignore
227 //				}
228 			if (soap) {
229 				SoapHeader soapHeader = ((SoapMessage) wsMessage).getSoapHeader();
230 				assertThat(soapHeader.getAttributeValue(QNameUtils.parseQNameString("foo"))).isNotNull();
231 				assertThat(soapHeader.getAttributeValue(QNameUtils.parseQNameString("foobar"))).isNotNull();
232 				assertThat(soapHeader.getAttributeValue(QNameUtils.parseQNameString("abaz"))).isNotNull();
233 				assertThat(soapHeader.getAttributeValue(QNameUtils.parseQNameString("bar"))).isNull();

247 			if (soap) {

250 				((SoapMessage) wsMessage).getSoapHeader().addAttribute(QNameUtils.parseQNameString("baz"), "baz");
251 			}

265 
266 		MessageChannel inputChannel = context.getBean(channelName, MessageChannel.class);
267 		Message<?> message =
268 				MessageBuilder.withPayload(payload).
269 						setHeader("foo", "foo").setHeader("foobar", "foobar").setHeader("abaz", "abaz").setHeader("bar", "bar").
270 						setHeader(WebServiceHeaders.SOAP_ACTION, "someAction").build();
271 		inputChannel.send(message);
272 		QueueChannel outputChannel = context.getBean("outputChannel", QueueChannel.class);
273 		return outputChannel.receive(0);

276 	private String extractStringResult(Message<?> replyMessage) throws Exception {
277 		Transformer transformer = TransformerFactory.newInstance().newTransformer();
278 		StringResult result = new StringResult();
279 		Object payload = replyMessage.getPayload();
280 		if (payload instanceof DOMSource) {
281 			transformer.transform(((DOMSource) replyMessage.getPayload()), result);
282 		}
283 		else if (payload instanceof Document) {
284 			transformer.transform(new DOMSource((Document) replyMessage.getPayload()), result);
285 		}
286 		else {
287 			throw new IllegalArgumentException("Unsupported payload type: " + payload.getClass().getName());
288 		}
289 		return result.toString();
290 	}

314 		public Object unmarshal(Source source) throws IOException, XmlMappingException {
315 			Element documentElement = (Element) ((DOMSource) source).getNode();
316 
317 			String name = DomUtils.getChildElementValueByTagName(documentElement, "name");
