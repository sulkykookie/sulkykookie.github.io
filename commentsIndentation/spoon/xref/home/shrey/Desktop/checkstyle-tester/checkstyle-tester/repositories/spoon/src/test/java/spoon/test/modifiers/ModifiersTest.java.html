50          spoon.addInputResource("./src/test/java/spoon/test/modifiers/testclasses/MethodVarArgs.java");
51          spoon.buildModel();

61          spoon.addInputResource("./src/test/java/spoon/test/modifiers/testclasses/MethodVarArgs.java");
62          spoon.getEnvironment().setShouldCompile(true);
63          spoon.run();
64      }

84      @Test
85      public void testSetVisibility() {
86          // contract: setVisibility should only work with public/private/protected modifiers
87  
88          Launcher spoon = new Launcher();
89          spoon.addInputResource("./src/test/java/spoon/test/modifiers/testclasses/StaticMethod.java");
90          spoon.buildModel();
91  
92          CtType<?> myClass = spoon.getFactory().Type().get(StaticMethod.class);
93          CtMethod methodPublicStatic = myClass.getMethodsByName("maMethod").get(0);
94  
95          assertEquals(ModifierKind.PUBLIC, methodPublicStatic.getVisibility());
96          methodPublicStatic.setVisibility(ModifierKind.PROTECTED);
97          assertEquals(ModifierKind.PROTECTED, methodPublicStatic.getVisibility());
98          try {
99              methodPublicStatic.setVisibility(ModifierKind.FINAL);
100             fail();
101         } catch (SpoonException e) {
102         }
103 
104         assertEquals(ModifierKind.PROTECTED, methodPublicStatic.getVisibility());
105     }

107     @Test
108     public void testGetModifiersHelpers() {
109         // contract: the CtModifiable helpers like isPublic, isFinal etc returns right values
110 
111         Launcher spoon = new Launcher();
112         spoon.addInputResource("./src/test/java/spoon/test/modifiers/testclasses/AbstractClass.java");
113         spoon.addInputResource("./src/test/java/spoon/test/modifiers/testclasses/ConcreteClass.java");
114         spoon.getEnvironment().setShouldCompile(true);
115         spoon.run();
116 
117         CtType<?> abstractClass = spoon.getFactory().Type().get(AbstractClass.class);
118 
119         checkCtModifiableHelpersAssertion(abstractClass, true, false, false, true, false, false);
120 
121         assertEquals(4, abstractClass.getFields().size());
122         for (CtField field : abstractClass.getFields()) {
123             switch (field.getSimpleName()) {
124                 case "privateField":
125                     checkCtModifiableHelpersAssertion(field, false, false, true, false, false, false);
126                     break;
127 
128                 case "protectedField":
129                     checkCtModifiableHelpersAssertion(field, false, true, false, false, false, false);
130                     break;
131 
132                 case "privateStaticField":
133                     checkCtModifiableHelpersAssertion(field, false, false, true, false, false, true);
134                     break;
135 
136                 case "publicFinalField":
137                     checkCtModifiableHelpersAssertion(field, true, false, false, false, true, false);
138                     break;
139 
140                 default:
141                     fail("The field "+field.getSimpleName()+" should be take into account.");
142             }
143         }
144 
145         assertEquals(4, abstractClass.getMethods().size());
146 
147         for (CtMethod method : abstractClass.getMethods()) {
148             switch (method.getSimpleName()) {
149                 case "method":
150                     checkCtModifiableHelpersAssertion(method, true, false, false, false, true, true);
151                     break;
152 
153                 case "onlyStatic":
154                     checkCtModifiableHelpersAssertion(method, true, false, false, false, false, true);
155                     break;
156 
157                 case "otherMethod":
158                     checkCtModifiableHelpersAssertion(method, false, true, false, true, false, false);
159                     break;
160 
161                 case "anotherOne":
162                     checkCtModifiableHelpersAssertion(method, false, false, false, true, false, false);
163                     break;
164 
165                 default:
166                     fail("The method "+method.getSimpleName()+" should be taken into account.");
167             }
168         }
169 
170         CtType<?> concreteClass = spoon.getFactory().Type().get("spoon.test.modifiers.testclasses.ConcreteClass");
171         checkCtModifiableHelpersAssertion(concreteClass, false, false, false, false, true, false);
172 
173         assertEquals(2, concreteClass.getFields().size());
174         for (CtField field : concreteClass.getFields()) {
175             switch (field.getSimpleName()) {
176                 case "className":
177                     checkCtModifiableHelpersAssertion(field, true, false, false, false, true, true);
178                     break;
179 
180                 case "test":
181                     checkCtModifiableHelpersAssertion(field, false, false, true, false, false, true);
182                     break;
183 
184                 default:
185                     fail("The field "+field.getSimpleName()+" should be take into account.");
186             }
187         }
188 
189         assertEquals(2, concreteClass.getMethods().size());
190         for (CtMethod method : concreteClass.getMethods()) {
191             switch (method.getSimpleName()) {
192                 case "otherMethod":
193                     checkCtModifiableHelpersAssertion(method, false, true, false, false, false, false);
194                     break;
195 
196                 case "anotherOne":
197                     checkCtModifiableHelpersAssertion(method, false, false, false, false, true, false);
198                     break;
199 
200                 default:
201                     fail("The method "+method.getSimpleName()+" should be taken into account.");
202             }
203         }
204     }
205 
206     private void checkCtModifiableHelpersAssertion(CtModifiable element, boolean isPublic, boolean isProtected, boolean isPrivate, boolean isAbstract, boolean isFinal, boolean isStatic) {
207         assertEquals("isPublic for "+element+" is wrong", isPublic, element.isPublic());
208         assertEquals("isProtected for "+element+" is wrong", isProtected, element.isProtected());
209         assertEquals("isPrivate for "+element+" is wrong", isPrivate, element.isPrivate());
210         assertEquals("isAbstract for "+element+" is wrong", isAbstract, element.isAbstract());
211         assertEquals("isFinal for "+element+" is wrong", isFinal, element.isFinal());
212         assertEquals("isStatic for "+element+" is wrong", isStatic, element.isStatic());
213     }

215     @Test
216     public void testClearModifiersByEmptySet() throws Exception {
217     	//contract: it is possible to remove modifiers by setModifiers(emptySet)
218     	CtType<?> ctClass = ModelUtils.buildClass(StaticMethod.class);
219     	assertTrue(ctClass.hasModifier(ModifierKind.PUBLIC));
220     	assertEquals(1, ctClass.getModifiers().size());
221     	
222     	ctClass.setModifiers(Collections.emptySet());
223     	assertFalse(ctClass.hasModifier(ModifierKind.PUBLIC));
224     	assertEquals(0, ctClass.getModifiers().size());
225     }

228     public void testClearModifiersByNull() throws Exception {
229     	//contract: it is possible to remove modifiers by setModifiers(null)
230     	CtType<?> ctClass = ModelUtils.buildClass(StaticMethod.class);
231     	assertTrue(ctClass.hasModifier(ModifierKind.PUBLIC));
232     	assertEquals(1, ctClass.getModifiers().size());
233 
234     	// contract: one can get the modifiers through CtRole.EMODIFIER
235         Collection<CtExtendedModifier> valueByRole = ctClass.getValueByRole(CtRole.EMODIFIER);
236         assertEquals(1, valueByRole.size());
237 
238         ctClass.setModifiers(null);
239     	assertFalse(ctClass.hasModifier(ModifierKind.PUBLIC));
240     	assertEquals(0, ctClass.getModifiers().size());
241 
242     }
