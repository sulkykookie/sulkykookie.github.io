57  /**
58   * <p>Builds the property panel for a use case.</p>
59   *
60   * <p>This is a type of Classifier, and like other Classifiers can have
61   *   attributes and operations (some processes use these to define
62   *   requirements). <em>Note</em>. ArgoUML does not currently support separate
63   *   compartments on the display for this.</p>
64   */
65  
66  public class PropPanelUseCase extends PropPanelClassifier {
67  
68      /**
69       * <p>Constructor. Builds up the various fields required.</p>
70       */
71      public PropPanelUseCase() {
72          // Invoke the Classifier constructor, but passing in our name and
73          // representation and requesting 3 columns
74          super("UseCase", ConfigLoader.getTabPropsOrientation());
75  
76          addField(Translator.localize("UMLMenu", "label.name"), getNameTextField());
77      	addField(Translator.localize("UMLMenu", "label.stereotype"), new UMLComboBoxNavigator(this, Translator.localize("UMLMenu", "tooltip.nav-stereo"), getStereotypeBox()));
78      	addField(Translator.localize("UMLMenu", "label.namespace"), getNamespaceComboBox());
79  
80  	PropPanelModifiers mPanel = new PropPanelModifiers(3);
81          Class mclass = (Class) ModelFacade.USE_CASE;
82  
83  	// since when do we know abstract usecases?
84  	//    mPanel.add("isAbstract", mclass, "isAbstract", "setAbstract",
85  	//               Translator.localize("UMLMenu", "checkbox.abstract-lc"), this);
86          mPanel.add("isLeaf", mclass, "isLeaf", "setLeaf",
87                  Translator.localize("UMLMenu", "checkbox.final-lc"), this);
88          mPanel.add("isRoot", mclass, "isRoot", "setRoot",
89                     localize("root"), this);
90  	addField(Translator.localize("UMLMenu", "label.modifiers"), mPanel);
91  
92  	JList extensionPoints = new UMLMutableLinkedList(new UMLUseCaseExtensionPointListModel(), null, ActionNewUseCaseExtensionPoint.SINGLETON);
93  	addField(Translator.localize("UMLMenu", "label.extension-points"),
94  		 new JScrollPane(extensionPoints));
95  
96  	addSeperator();
97  
98  	addField(Translator.localize("UMLMenu", "label.generalizations"), getGeneralizationScroll());
99  	addField(Translator.localize("UMLMenu", "label.specializations"), getSpecializationScroll());
100 
101 	JList extendsList = new UMLLinkedList(new UMLUseCaseExtendListModel());
102 	addField(Translator.localize("UMLMenu", "label.extends"),
103 		 new JScrollPane(extendsList));
104 
105 	JList includesList = new UMLLinkedList(new UMLUseCaseIncludeListModel());
106 	addField(Translator.localize("UMLMenu", "label.includes"),
107 		 new JScrollPane(includesList));
108 
109 	addSeperator();
110 
111         addField(Translator.localize("UMLMenu", "label.association-ends"), 
112             getAssociationEndScroll());
113 
114         
115 
116         new PropPanelButton(this, buttonPanel, _navUpIcon,
117                 Translator.localize("UMLMenu", "button.go-up"), "navigateNamespace",
118                             null);
119         new PropPanelButton(this, buttonPanel, _useCaseIcon,
120                 Translator.localize("UMLMenu", "button.add-usecase"), "newUseCase",
121                             null);
122         new PropPanelButton(this, buttonPanel, _extensionPointIcon,
123                             localize("Add extension point"),
124                             "newExtensionPoint",
125                             null);
126         new PropPanelButton(this, buttonPanel, _deleteIcon,
127                             localize("Delete"), "removeElement",
128                             null);
129 
130     }
131 
132 
133     /**
134      * <p>Invoked by the "Add use case" toolbar button to create a new use case
135      *   property panel in the same namespace as the current use case.</p>
136      *
137      * <p>This code uses getFactory and adds the use case explicitly to the
138      *   namespace. Extended to actually navigate to the new use case.</p>
139      */
140 
141     public void newUseCase() {
142         Object target = getTarget();
143 
144         if (ModelFacade.isAUseCase(target)) {
145             Object ns = ModelFacade.getNamespace(target);
146 
147             if (ns != null) {
148                 Object useCase = UseCasesFactory.getFactory().createUseCase();
149                 ModelFacade.addOwnedElement(ns, useCase);
150                 TargetManager.getInstance().setTarget(useCase);
151             }
152         }
153     }
154 
155 
156     /**
157      * <p>Invoked by the "Add extension point" toolbar button to create a new
158      *   extension point for this use case in the same namespace as the current
159      *   use case.</p>
160      *
161      * <p>This code uses getFactory and adds the extension point explicitly to
162      *   the, making its associated use case the current use case.</p>
163      */
164     public void newExtensionPoint() {
165         Object target = getTarget();
166 
167         if (ModelFacade.isAUseCase(target)) {
168             TargetManager.getInstance().setTarget(UseCasesFactory.getFactory().buildExtensionPoint(target));
169         }
170     }
171 } /* end class PropPanelUseCase */
