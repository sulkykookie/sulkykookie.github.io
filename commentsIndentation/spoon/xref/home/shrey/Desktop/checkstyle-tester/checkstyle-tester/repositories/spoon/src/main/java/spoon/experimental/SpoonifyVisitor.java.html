80  	private boolean isLeafTypeReference(CtElement element) {
81  		if (!(element instanceof CtTypeReference)) {
82  			return false;
83  		}
84  		if (element instanceof CtArrayTypeReference
85  				|| element instanceof CtWildcardReference
86  				|| element instanceof CtTypeParameterReference
87  				|| element instanceof CtIntersectionTypeReference) {
88  			return false;
89  		}
90  		CtTypeReference reference = (CtTypeReference) element;
91  		return reference.getDeclaringType() == null
92  				&& reference.getActualTypeArguments().isEmpty()
93  				&& reference.getAnnotations().isEmpty()
94  				&& reference.getComments().isEmpty();
95  	}

98  	public void enter(CtElement element) {
99  		if (element instanceof CtPackageReference
100 				&& isLeafTypeReference(element.getParent())) {
101 			return;
102 		}
103 		tabs++;
104 
105 		String elementClass = element.getClass().getSimpleName();
106 		if (elementClass.endsWith("Impl")) {
107 			elementClass = elementClass.replace("Impl", "");
108 		}
109 
110 		String variableName = null;
111 		if (isLeafTypeReference(element)) {
112 			CtTypeReference typeRef = (CtTypeReference) element;
113 			if (typeRef.isPrimitive()) {
114 				switch (typeRef.getSimpleName()) {
115 					case "int":
116 						variableName = "factory.Type().INTEGER_PRIMITIVE";
117 						break;
118 					case "char":
119 						variableName = "factory.Type().CHARACTER_PRIMITIVE";
120 						break;
121 					default:
122 						variableName = "factory.Type()." + typeRef.getSimpleName().toUpperCase() + "_PRIMITIVE";
123 				}
124 			} else if (typeRef.getSimpleName().equals("<nulltype>")) {
125 				variableName = "factory.Type().NULL_TYPE";
126 			} else if (typeRef.getPackage().isImplicit()) {
127 				variableName =  "factory.Type().createSimplyQualifiedReference(\"" + typeRef.getQualifiedName() + "\")";
128 			} else {
129 				variableName =  "factory.Type().createReference(\"" + typeRef.getQualifiedName() + "\")";
130 			}
131 		} else {
132 			variableName = getVariableName(elementClass);
133 			result.append(printTabs() + elementClass + " " + variableName + " = factory.create" + elementClass.replaceFirst("Ct", "") + "();");
134 			result.append("\n");
135 
136 			if (element.isImplicit()) {
137 				result.append(printTabs() + variableName + ".setImplicit(true);\n");
138 			}
139 			propertyScanner.variableName = variableName;
140 			element.accept(propertyScanner);
141 		}
142 
143 		if (element.isParentInitialized() && !parentName.isEmpty()) {
144 			CtRole elementRoleInParent = element.getRoleInParent();
145 
146 			CtElement parent = element.getParent();
147 			Object o  = parent.getValueByRole(elementRoleInParent);
148 			if (o instanceof Map) {
149 				handleContainer(element, parent, elementRoleInParent, variableName, "Map");
150 			} else if (o instanceof List) {
151 				handleContainer(element, parent, elementRoleInParent, variableName, "List");
152 			} else if (o instanceof Set) {
153 				handleContainer(element, parent, elementRoleInParent, variableName, "Set");
154 			} else {
155 				result.append(printTabs() + parentName.peek() + ".setValueByRole(CtRole." + elementRoleInParent.name() + ", " + variableName + ");\n");
156 			}
157 		}
158 
159 		parentName.push(variableName);
160 		roleContainer.push(new HashMap<>());
161 	}
162 
163 	private String getVariableName(String className) {
164 		if (!variableCount.containsKey(className)) {
165 			variableCount.put(className, 0);
166 		}
167 		int count = variableCount.get(className);
168 		String variableName = className.substring(0, 1).toLowerCase() + className.substring(1) + count;
169 		variableCount.put(className, count + 1);
170 		return variableName;
171 	}
172 
173 	private void handleContainer(CtElement element, CtElement parent, CtRole elementRoleInParent, String variableName, String container) {
174 		String concreteClass = null;
175 
176 		switch (container) {
177 			case "Map":
178 				concreteClass = "HashMap";
179 				break;
180 			case "List":
181 				concreteClass = "ArrayList";
182 				break;
183 			case "Set":
184 				concreteClass = "HashSet";
185 				break;
186 		}
187 
188 		String containerName;
189 		if (!roleContainer.peek().containsKey(elementRoleInParent)) {
190 			containerName = parentName.peek() + elementRoleInParent.toString().substring(0, 1).toUpperCase() + elementRoleInParent.toString().substring(1) + "s";
191 			roleContainer.peek().put(elementRoleInParent, containerName);
192 			result.append(printTabs() + container + " " + containerName + " = new " + concreteClass + "();\n");
193 		} else {
194 			containerName = roleContainer.peek().get(elementRoleInParent);
195 		}
196 
197 		if (container.equals("Map")) {
198 			//This is going to be dirty.
199 			//In case where different keys point toward the same value,
200 			//some useless variable will be created.
201 			List<String> keys = new ArrayList<>();
202 			Map m = parent.getValueByRole(elementRoleInParent);
203 			for (Object e : m.entrySet()) {
204 				Map.Entry entry = (Map.Entry) e;
205 				if (entry.getValue().equals(element)) {
206 					keys.add((String) entry.getKey());
207 				}
208 			}
209 			for (String key: keys) {
210 				result.append(printTabs() + containerName + ".put(\"" + key + "\", " + variableName + ");\n");
211 			}
212 
213 		} else {
214 			result.append(printTabs() + containerName + ".add(" + variableName + ");\n");
215 		}
216 	}
217 
218 	public void exit(CtElement element) {
219 		if (element instanceof CtPackageReference
220 				&& isLeafTypeReference(element.getParent())) {
221 			return;
222 		}
223 		if (!roleContainer.peek().isEmpty()) {
224 			for (CtRole role: roleContainer.peek().keySet()) {
225 				String variableName = roleContainer.peek().get(role);
226 				result.append(printTabs() + parentName.peek() + ".setValueByRole(CtRole." + role.name() + ", " + variableName + ");\n");
227 			}
228 		}
229 		parentName.pop();
230 		roleContainer.pop();
231 		tabs--;
232 	}
233 
234 	private String printTabs() {
235 		String res = "";
236 		if (printTabs) {
237 			for (int i = 0; i < tabs; i++) {
238 				res += "\t";
239 			}
240 		}
241 		return res;
242 	}

244 	class PropertyScanner extends CtInheritanceScanner {
245 		public String variableName;
246 
247 		/**
248 		 * visitLiteral
249 		 * @param element
250 		 */
251 		@Override
252 		public void visitCtLiteral(CtLiteral element) {
253 			if (element.getType().isPrimitive()) {
254 				result.append(printTabs() + variableName + ".setValue((" + element.getType().getSimpleName() + ") " + element.toString() + ");\n");
255 				if (element.getBase() != null) {
256 					result.append(printTabs() + variableName + ".setBase(LiteralBase." + element.getBase().name() + ");\n");
257 				}
258 			} else if (element.getType().getQualifiedName().equals("java.lang.String")) {
259 				result.append(printTabs() + variableName + ".setValue(\"" + StringEscapeUtils.escapeJava((String) element.getValue()) + "\");\n");
260 			}
261 			super.visitCtLiteral(element);
262 		}
263 
264 		@Override
265 		public void visitCtBinaryOperator(CtBinaryOperator element) {
266 			result.append(printTabs() + variableName + ".setKind(BinaryOperatorKind." + element.getKind().name() + ");\n");
267 			super.visitCtBinaryOperator(element);
268 		}
269 
270 		@Override
271 		public void visitCtUnaryOperator(CtUnaryOperator element) {
272 			result.append(printTabs() + variableName + ".setKind(UnaryOperatorKind." + element.getKind().name() + ");\n");
273 			super.visitCtUnaryOperator(element);
274 		}
275 
276 		@Override
277 		public void visitCtOperatorAssignment(CtOperatorAssignment element) {
278 			result.append(printTabs() + variableName + ".setKind(BinaryOperatorKind." + element.getKind().name() + ");\n");
279 			super.visitCtOperatorAssignment(element);
280 		}
281 
282 		@Override
283 		public void visitCtComment(CtComment element) {
284 			result.append(printTabs() + variableName + ".setCommentType(CtComment.CommentType." + element.getCommentType().name() + ");\n");
285 			result.append(printTabs() + variableName + ".setContent(\"" + StringEscapeUtils.escapeJava(element.getContent()) + "\");\n");
286 			super.visitCtComment(element);
287 		}
288 
289 		@Override
290 		public void visitCtParameter(CtParameter element) {
291 			if (element.isVarArgs()) {
292 				result.append(printTabs() + variableName + ".setVarArgs(true);\n");
293 			}
294 			super.visitCtParameter(element);
295 		}
296 
297 		@Override
298 		public void visitCtMethod(CtMethod element) {
299 			if (element.isDefaultMethod()) {
300 				result.append(printTabs() + variableName + ".setDefaultMethod(true);\n");
301 			}
302 			super.visitCtMethod(element);
303 		}
304 
305 		@Override
306 		public void scanCtReference(CtReference element) {
307 			result.append(printTabs() + variableName + ".setSimpleName(\"" + element.getSimpleName() + "\");\n");
308 		}
309 
310 		@Override
311 		public void scanCtNamedElement(CtNamedElement element) {
312 			result.append(printTabs() + variableName + ".setSimpleName(\"" + element.getSimpleName() + "\");\n");
313 		}
314 
315 		@Override
316 		public void scanCtModifiable(CtModifiable element) {
317 			if (!element.getModifiers().isEmpty()) {
318 				result.append(printTabs() + "Set<ModifierKind> " + variableName + "Modifiers = new HashSet<>();\n");
319 				for (ModifierKind mod : element.getModifiers()) {
320 					result.append(printTabs() + variableName + "Modifiers.add(ModifierKind." + mod.name() + ");\n");
321 				}
322 				result.append(printTabs() + variableName + ".setModifiers(" + variableName + "Modifiers);\n");
323 			}
324 		}
325 
326 		@Override
327 		public void scanCtStatement(CtStatement element) {
328 			if (element.getLabel() != null) {
329 				result.append(printTabs() + variableName + ".setLabel(\"" + element.getLabel() + "\");\n");
330 			}
331 		}
332 
333 		@Override
334 		public void scanCtLabelledFlowBreak(CtLabelledFlowBreak element) {
335 			if (element.getTargetLabel() != null) {
336 				result.append(printTabs() + variableName + ".setTargetLabel(\"" + element.getTargetLabel() + "\");\n");
337 			}
338 		}
339 	}
