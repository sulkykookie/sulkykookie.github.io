59  		SpoonAPI spoon = new Launcher();
60  		spoon.addInputResource("src/main/java/spoon/reflect/factory");
61  		spoon.buildModel();
62  
63  		for (CtType t : spoon.getFactory().Package().getRootPackage().getElements(new AbstractFilter<CtType>() {
64  			@Override
65  			public boolean matches(CtType element) {
66  				return super.matches(element)
67  						&& element.getSimpleName().contains("Factory");
68  			}
69  		})) {
70  			for (Object o : t.getFields()) {
71  				CtField f = (CtField) o;
72  				if ("factory".equals(f.getSimpleName())) {
73  					continue;
74  				}
75  				if (f.hasModifier(ModifierKind.FINAL) || f.hasModifier(ModifierKind.TRANSIENT)) {
76  					continue;
77  				}
78  
79  				fail("architectural constraint: a factory must be stateless");
80  			}
81  		}

84  	@Test
85  	public void testFactorySubFactory() {
86  		// contract:: all subfactory methods must also be in the main factory
87  		// this is very important for usability and discoverability
88  		final Launcher launcher = new Launcher();
89  		launcher.addInputResource("./src/main/java/spoon/reflect/factory");
90  		class SanityCheck { int val = 0; }
91  		SanityCheck sanityCheck = new SanityCheck();
92  		launcher.addProcessor(new AbstractManualProcessor() {
93  			@Override
94  			public void process() {
95  				CtType factoryImpl = getFactory().Interface().get(Factory.class);
96  				CtPackage factoryPackage = getFactory().Package().getOrCreate("spoon.reflect.factory");
97  				CtInterface itf = getFactory().Interface().create("MegaFactoryItf");
98  				CtClass impl = getFactory().Class().create("MegaFactory");
99  				for (CtType<?> t : factoryPackage.getTypes()) {
100 					if (t.getSimpleName().startsWith("Mega")) {
101 						continue;
102 					}
103 
104 					for (CtMethod<?> m : t.getMethods()) {
105 						// we check only public methods
106 						if (m.hasModifier(ModifierKind.PUBLIC) == false) {
107 							continue;
108 						}
109 						// we only consider factory methods
110 						if (!m.getSimpleName().startsWith("create")) {
111 							continue;
112 						}
113 
114 						// too generic, what should we create??
115 						if ("create".equals(m.getSimpleName())) {
116 							String simpleNameType = m.getType().getSimpleName().replace("Ct", "");
117 							CtMethod method = m.clone();
118 
119 							method.setSimpleName("create" + simpleNameType);
120 							assertTrue(method.getSignature() + " (from " + t.getQualifiedName() + ") is not present in the main factory", factoryImpl.hasMethod(method));
121 							continue;
122 						}
123 
124 						// too generic, is it a fieldref? an execref? etc
125 						if ("createReference".equals(m.getSimpleName())) {
126 							continue;
127 						}
128 
129 						if (m.getModifiers().contains(ModifierKind.ABSTRACT)) {
130 							continue;
131 						}
132 
133 						sanityCheck.val++;
134 
135 						// the core assertion
136 						assertTrue(m.getSignature() + " is not present in the main factory", factoryImpl.hasMethod(m));
137 					}
138 				}
139 			}
140 		});
141 		launcher.run();
142 		assertTrue(sanityCheck.val > 100);
143 	}

147 	@Test
148 	public void testSrcMainJava() {
149 		Launcher spoon = new Launcher();
150 		spoon.getEnvironment().setCommentEnabled(true);
151 		spoon.addInputResource("src/main/java/");
152 
153 		// contract: all non-trivial public methods should be documented with proper API Javadoc
154 		spoon.buildModel();
155 		List<String> notDocumented = new ArrayList<>();
156 		for (CtMethod method : spoon.getModel().getElements(new TypeFilter<>(CtMethod.class))) {
157 
158 			// now we see whether this should be documented
159 			if (method.hasModifier(ModifierKind.PUBLIC) // public methods should be documented
160 					&& !method.getSimpleName().startsWith("get") // all kinds of setters can be undocumented
161 					&& !method.getSimpleName().startsWith("set")
162 					&& !method.getSimpleName().startsWith("is")
163 					&& !method.getSimpleName().startsWith("add")
164 					&& !method.getSimpleName().startsWith("remove")
165 					&& method.getTopDefinitions().isEmpty() // only the top declarations should be documented (not the overriding methods which are lower in the hierarchy)
166 					&& (
167 							method.hasModifier(ModifierKind.ABSTRACT) // all interface methods and abstract class methods must be documented
168 
169 							// GOOD FIRST ISSUE
170 							// ideally we want that **all** public methods are documented
171 							// so far, we have this arbitrary limit in the condition below (35)
172 							// because it's a huge task to document everything at once
173 							// so to contribute to Spoon, what you can do is
174 							// 1) you lower the threshold (eg 33)
175 							// 2) you run test `documentedTest`, it will output a list on undocumented methods
176 							// 3) you document those methods
177 							// 4) you run the test again to check that it passes
178 							// 4) you commit your changes and create the corresponding pull requests
179 							|| method.filterChildren(new TypeFilter<>(CtCodeElement.class)).list().size() > 33  // means that only large methods must be documented
180 			)) {
181 
182 				// OK it should be properly documented
183 
184 				// is it really well documented?
185 				if (method.getDocComment().length() <= 15) { // the Javadoc must be at least at least 15 characters (still pretty short...)
186 					notDocumented.add(method.getParent(CtType.class).getQualifiedName() + "#" + method.getSignature());
187 				}
188 			}
189 		}
190 		if (!notDocumented.isEmpty()) {
191 			fail(notDocumented.size() + " public methods should be documented with proper API documentation: \n" + StringUtils.join(notDocumented, "\n"));
192 		}
193 
194 		// contract: Spoon's code never uses TreeSet constructor, because they implicitly depend on Comparable (no static check, only dynamic checks)
195 		List<CtConstructorCall> treeSetWithoutComparators = spoon.getFactory().Package().getRootPackage().filterChildren(new AbstractFilter<CtConstructorCall>() {
196 			@Override
197 			public boolean matches(CtConstructorCall element) {
198 				return element.getType().getActualClass().equals(TreeSet.class) && element.getArguments().isEmpty();
199 			}
200 		}).list();
201 
202 		assertEquals(0, treeSetWithoutComparators.size());
203 	}

432 	public static void assertSetEquals(String msg, Set<?> set1, Set<?> set2) {
433 		if (set1 == null || set2 == null) {
434 			throw new IllegalArgumentException();
435 		}
436 
437 		if (set1.size() != set2.size()) {
438 			throw new AssertionError(msg + "\n\nDetails: " + computeDifference(set1, set2));
439 		}
440 
441 		if (!set1.containsAll(set2)) {
442 			throw new AssertionError(msg + "\n\nDetails: " + computeDifference(set1, set2));
443 		}
444 	}
445 
446 	private static String computeDifference(Set<?> set1, Set<?> set2) {
447 		Set<String> results = new HashSet<>();
448 
449 		for (Object o : set1) {
450 			if (!set2.contains(o)) {
451 				results.add("Missing package " + o + " in computed set");
452 			} else {
453 				set2.remove(o);
454 			}
455 		}
456 
457 		for (Object o : set2) {
458 			results.add("Package " + o + " presents in computed but not expected set.");
459 		}
460 		return StringUtils.join(results, "\n");
461 	}
