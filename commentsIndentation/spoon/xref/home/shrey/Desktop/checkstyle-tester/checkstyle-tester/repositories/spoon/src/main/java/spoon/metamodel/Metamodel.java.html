208 	/**
209 	 * @return Spoon {@link Metamodel}, which is built once and then returns cached version
210 	 */
211 	public static Metamodel getInstance() {
212 		if (instance == null) {
213 			try {
214 				//this is needed just for CtGenerationTest#testGenerateRoleHandler
215 				//which must not use RoleHandler at time when RoleHandler is generated and Spoon model doesn't fit to old RoleHandlers
216 				//to avoid egg/chicken problem
217 				if ("true".equals(System.getProperty(MetamodelProperty.class.getName() + "-noRoleHandler"))) {
218 					MetamodelProperty.useRuntimeMethodInvocation = true;
219 				}
220 			} catch (SecurityException e) {
221 				//ignore that
222 			}
223 			instance = new Metamodel();
224 		}
225 		return instance;
226 	}

241 	/**
242 	 * @param factory already loaded factory with all Spoon model types
243 	 */
244 	protected Metamodel(Factory factory) {
245 		for (String apiPackage : MODEL_IFACE_PACKAGES) {
246 			if (factory.Package().get(apiPackage) == null) {
247 				throw new SpoonException("Spoon Factory model is missing API package " + apiPackage);
248 			}
249 			String implPackage = replaceApiToImplPackage(apiPackage);
250 			if (factory.Package().get(implPackage) == null) {
251 				throw new SpoonException("Spoon Factory model is missing implementation package " + implPackage);
252 			}
253 		}
254 
255 		//search for all interfaces of spoon model and create MetamodelConcepts for them
256 		factory.getModel().filterChildren(new TypeFilter<>(CtInterface.class))
257 			.forEach((CtInterface<?> iface) -> {
258 				if (MODEL_IFACE_PACKAGES.contains(iface.getPackage().getQualifiedName())) {
259 					getOrCreateConcept(iface);
260 				}
261 			});
262 	}

264 	/**
265 	 * Creates a {@link Metamodel} in runtime mode when spoon sources are not available.
266 	 *
267 	 * See also {@link #getInstance()}.
268 	 */
269 	private Metamodel() {
270 		for (CtType<?> iface : getAllMetamodelInterfaces()) {
271 			if (iface instanceof CtInterface) {
272 				getOrCreateConcept(iface);
273 			}
274 		}
275 	}

339 		return (CtClass<?>) getType(impl, iface);

342 	/**
343 	 * @param impl the implementation class of a Spoon element
344 	 * @return {@link CtInterface} of Spoon model which represents API of the spoon model class. null if there is no implementation.
345 	 */
346 	public static CtInterface<?> getInterfaceOfImplementation(CtClass<?> impl) {
347 		String iface = impl.getQualifiedName();
348 		if (iface.endsWith(CLASS_SUFFIX) == false || iface.startsWith("spoon.support.reflect.") == false) {
349 			throw new SpoonException("Unexpected spoon model implementation class: " + impl.getQualifiedName());
350 		}
351 		iface = iface.substring(0, iface.length() - CLASS_SUFFIX.length());
352 		iface = iface.replace("spoon.support.reflect", "spoon.reflect");
353 		return (CtInterface<?>) getType(iface, impl);
354 	}

356 	/**
357 	 * @param method to be checked method
358 	 * @return {@link CtRole} of spoon model method. Looking into all super class/interface implementations of this method
359 	 */
360 	public static CtRole getRoleOfMethod(CtMethod<?> method) {
361 		Factory f = method.getFactory();
362 		CtAnnotation<PropertyGetter> getter = getInheritedAnnotation(method, f.createCtTypeReference(PropertyGetter.class));
363 		if (getter != null) {
364 			return getter.getActualAnnotation().role();
365 		}
366 		CtAnnotation<PropertySetter> setter = getInheritedAnnotation(method, f.createCtTypeReference(PropertySetter.class));
367 		if (setter != null) {
368 			return setter.getActualAnnotation().role();
369 		}
370 		return null;
371 	}

373 	private static CtType<?> getType(String qualifiedName, CtElement anElement) {
374 		Class aClass;
375 		try {
376 			aClass = anElement.getClass().getClassLoader().loadClass(qualifiedName);
377 		} catch (ClassNotFoundException e) {
378 			//OK, that interface has no implementation class
379 			return null;
380 		}
381 		return anElement.getFactory().Type().get(aClass);
382 	}

385 	private static final String modelApiImplPackage = "spoon.support.reflect";

387 	private static String replaceApiToImplPackage(String modelInterfaceQName) {
388 		if (modelInterfaceQName.startsWith(modelApiPackage) == false) {
389 			throw new SpoonException("The qualified name " + modelInterfaceQName + " doesn't belong to Spoon model API package: " + modelApiPackage);
390 		}
391 		return modelApiImplPackage + modelInterfaceQName.substring(modelApiPackage.length());
392 	}

394 	private static Factory createFactory(File spoonJavaSourcesDirectory) {
395 		final Launcher launcher = new Launcher();
396 		launcher.getEnvironment().setNoClasspath(true);
397 		launcher.getEnvironment().setCommentEnabled(true);
398 //		// Spoon model interfaces
399 		Arrays.asList("spoon/reflect/code",
400 				"spoon/reflect/declaration",
401 				"spoon/reflect/reference",
402 				"spoon/support/reflect/declaration",
403 				"spoon/support/reflect/code",
404 				"spoon/support/reflect/reference").forEach(path ->
405 			launcher.addInputResource(new FileSystemFolder(new File(spoonJavaSourcesDirectory, path))));
406 		launcher.buildModel();
407 		return launcher.getFactory();
408 	}

410 	/**
411 	 * @param type can be class or interface of Spoon model element
412 	 * @return existing or creates and initializes new {@link MetamodelConcept} which represents the `type`
413 	 */
414 	private MetamodelConcept getOrCreateConcept(CtType<?> type) {
415 		String conceptName = getConceptName(type);
416 		return getOrCreate(nameToConcept, conceptName,
417 				() -> new MetamodelConcept(conceptName),
418 				mmConcept -> initializeConcept(type, mmConcept));
419 	}
420 
421 	/**
422 	 * is called once for each {@link MetamodelConcept}, to initialize it.
423 	 * @param type a class or inteface of the spoon model element
424 	 * @param mmConcept to be initialize {@link MetamodelConcept}
425 	 */
426 	private void initializeConcept(CtType<?> type, MetamodelConcept mmConcept) {
427 		//it is not initialized yet. Do it now
428 		if (type instanceof CtInterface<?>) {
429 			CtInterface<?> iface = (CtInterface<?>) type;
430 			mmConcept.setModelClass(getImplementationOfInterface(iface));
431 			mmConcept.setModelInterface(iface);
432 		} else if (type instanceof CtClass<?>) {
433 			CtClass<?> clazz = (CtClass<?>) type;
434 			mmConcept.setModelClass(clazz);
435 			mmConcept.setModelInterface(getInterfaceOfImplementation(clazz));
436 		} else {
437 			throw new SpoonException("Unexpected spoon model type: " + type.getQualifiedName());
438 		}
439 
440 		//add fields of interface
441 		if (mmConcept.getMetamodelInterface() != null) {
442 			//add fields of interface too. They are not added by above call of addFieldsOfType, because the MetamodelConcept already exists in nameToConcept
443 			addFieldsOfType(mmConcept, mmConcept.getMetamodelInterface());
444 		}
445 		//initialize all fields
446 		mmConcept.getRoleToProperty().forEach((role, mmField) -> {
447 			//if there are more methods for the same field then choose the one which best matches the field type
448 			mmField.sortByBestMatch();
449 			//finally initialize value type of this field
450 			mmField.setValueType(mmField.detectValueType());
451 		});
452 	}

456 	 * @param mmConcept the owner of to be created fields
457 	 * @param ctType to be scanned {@link CtType}
458 	 */
459 	private void addFieldsOfType(MetamodelConcept mmConcept, CtType<?> ctType) {
460 		ctType.getTypeMembers().forEach(typeMember -> {
461 			if (typeMember instanceof CtMethod<?>) {
462 				CtMethod<?> method = (CtMethod<?>) typeMember;
463 				CtRole role = getRoleOfMethod(method);
464 				if (role != null) {
465 					MetamodelProperty field = mmConcept.getOrCreateMMField(role);
466 					field.addMethod(method);
467 				} else {
468 					mmConcept.otherMethods.add(method);
469 				}
470 			}
471 		});
472 		addFieldsOfSuperType(mmConcept, ctType.getSuperclass());
473 		ctType.getSuperInterfaces().forEach(superIfaceRef -> addFieldsOfSuperType(mmConcept, superIfaceRef));
474 	}

488 	/**
489 	 * add all fields of `superTypeRef` into `mmConcept`
490 	 * @param concept sub type
491 	 * @param superTypeRef super type
492 	 */
493 	private void addFieldsOfSuperType(MetamodelConcept concept, CtTypeReference<?> superTypeRef) {
494 		if (superTypeRef == null) {
495 			return;
496 		}
497 		if (EXPECTED_TYPES_NOT_IN_CLASSPATH.contains(superTypeRef.getQualifiedName())) {
498 			//ignore classes which are not part of spoon model
499 			return;
500 		}
501 		CtType<?> superType = superTypeRef.getTypeDeclaration();
502 		if (superType == null) {
503 			throw new SpoonException("Cannot create spoon meta model. The class " + superTypeRef.getQualifiedName() + " is missing class path");
504 		}
505 		//call getOrCreateConcept recursively for super concepts
506 		MetamodelConcept superConcept = getOrCreateConcept(superType);
507 		if (superConcept != concept) {
508 			concept.addSuperConcept(superConcept);
509 		}
510 	}

512 	static <K, V> V getOrCreate(Map<K, V> map, K key, Supplier<V> valueCreator) {
513 		return getOrCreate(map, key, valueCreator, null);
514 	}
515 	/**
516 	 * @param initializer is called immediately after the value is added to the map
517 	 */
518 	static <K, V> V getOrCreate(Map<K, V> map, K key, Supplier<V> valueCreator, Consumer<V> initializer) {
519 		V value = map.get(key);
520 		if (value == null) {
521 			value = valueCreator.get();
522 			map.put(key, value);
523 			if (initializer != null) {
524 				initializer.accept(value);
525 			}
526 		}
527 		return value;
528 	}
529 	static <T> boolean addUniqueObject(Collection<T> col, T o) {
530 		if (containsObject(col, o)) {
531 			return false;
532 		}
533 		col.add(o);
534 		return true;
535 	}
536 	static boolean containsObject(Iterable<?> iter, Object o) {
537 		for (Object object : iter) {
538 			if (object == o) {
539 				return true;
540 			}
541 		}
542 		return false;
543 	}

546 	/**
547 	 * @param method a start method
548 	 * @param annotationType a searched annotation type
549 	 * @return annotation from the first method in superClass and superInterface hierarchy for the method with required annotationType
550 	 */
551 	private static <A extends Annotation> CtAnnotation<A> getInheritedAnnotation(CtMethod<?> method, CtTypeReference<A> annotationType) {
552 		CtAnnotation<A> annotation = method.getAnnotation(annotationType);
553 		if (annotation == null) {
554 			CtType<?> declType = method.getDeclaringType();
555 			final ClassTypingContext ctc = new ClassTypingContext(declType);
556 			annotation = declType.map(new AllTypeMembersFunction(CtMethod.class)).map((CtMethod<?> currentMethod) -> {
557 				if (method == currentMethod) {
558 					return null;
559 				}
560 				if (ctc.isSameSignature(method, currentMethod)) {
561 					CtAnnotation<A> annotation2 = currentMethod.getAnnotation(annotationType);
562 					if (annotation2 != null) {
563 						return annotation2;
564 					}
565 				}
566 				return null;
567 			}).first();
568 		}
569 		return annotation;
570 	}
