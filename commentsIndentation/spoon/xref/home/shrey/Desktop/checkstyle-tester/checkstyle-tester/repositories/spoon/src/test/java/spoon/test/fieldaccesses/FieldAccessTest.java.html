196 		CtType<?> type = build("spoon.test.fieldaccesses.testclasses", "TargetedAccessPosition");
197 		List<Object> list1 = type.filterChildren(new TypeFilter<>(CtFieldWrite.class)).list();
198 		assertEquals(1, list1.size());

201 		assertEquals(3, list2.size());
202 
203 		// this is the tricky part :-)
204 		Collections.reverse(list2);
205 
206 		// now we replace them all
207 		for(CtFieldRead r : list2) {
208 			// quick'n'dirty solution with snippets
209 			// a better way is to create a CtInvocation
210 			CtCodeSnippetExpression s = r.getFactory().createCodeSnippetExpression("f("+r.toString()+")");

212 		}
213 
214 		assertEquals("f(f(f(t.ta).ta).ta).ta = t", type.getMethodsByName("foo").get(0).getBody().getStatement(1).toString());

280 		// contract: When we use var++, the variable is a read access with an unary operator.
281 		final CtType<Panini> aMole = buildClass(Panini.class);
282 		final CtMethod<?> make = aMole.getMethodsByName("make").get(0);
283 		final List<CtUnaryOperator<?>> unaryOperators = make.getElements(new TypeFilter<>(CtUnaryOperator.class));
284 
285 		final CtFieldWrite<Object> fieldRead = aMole.getFactory().Core().createFieldWrite();
286 		fieldRead.setTarget(aMole.getFactory().Code().createThisAccess(aMole.getReference(), true));
287 		final CtFieldReference fieldReference = aMole.getField("i").getReference();
288 		fieldRead.setVariable(fieldReference);
289 
290 		assertEquals(2, unaryOperators.size());
291 		final CtUnaryOperator<?> first = unaryOperators.get(0);
292 		assertEquals(UnaryOperatorKind.POSTINC, first.getKind());
293 		assertEquals(fieldRead, first.getOperand());
294 		assertEquals("i++", first.toString());
295 
296 		final CtUnaryOperator<?> second = unaryOperators.get(1);
297 		assertEquals(UnaryOperatorKind.PREINC, second.getKind());
298 		assertEquals(fieldRead, second.getOperand());
299 		assertEquals("++i", second.toString());
300 	}

448 		CtMethod method = mouse.filterChildren((CtMethod m)->"meth1".equals(m.getSimpleName())).first();
449 		
450 		CtFieldReference ageFR = method.filterChildren((CtFieldReference fr)->"age".equals(fr.getSimpleName())).first();
