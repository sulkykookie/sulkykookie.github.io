213   enum BalanceSwitchMode {
214     SYNC,
215     ASYNC
216   }
217 
218   /**
219    * Assigns balancer switch according to BalanceSwitchMode
220    * @param b new balancer switch
221    * @param mode BalanceSwitchMode
222    * @return old balancer switch
223    */
224   boolean switchBalancer(final boolean b, BalanceSwitchMode mode) throws IOException {
225     boolean oldValue = master.loadBalancerTracker.isBalancerOn();
226     boolean newValue = b;
227     try {
228       if (master.cpHost != null) {
229         newValue = master.cpHost.preBalanceSwitch(newValue);
230       }
231       try {
232         if (mode == BalanceSwitchMode.SYNC) {
233           synchronized (master.balancer) {
234             master.loadBalancerTracker.setBalancerOn(newValue);
235           }
236         } else {
237           master.loadBalancerTracker.setBalancerOn(newValue);
238         }
239       } catch (KeeperException ke) {
240         throw new IOException(ke);
241       }
242       LOG.info(master.getClientIdAuditPrefix() + " set balanceSwitch=" + newValue);
243       if (master.cpHost != null) {
244         master.cpHost.postBalanceSwitch(oldValue, newValue);
245       }
246     } catch (IOException ioe) {
247       LOG.warn("Error flipping balance switch", ioe);
248     }
249     return oldValue;
250   }
251 
252   boolean synchronousBalanceSwitch(final boolean b) throws IOException {
253     return switchBalancer(b, BalanceSwitchMode.SYNC);
254   }

285   @Override
286   @QosPriority(priority=HConstants.ADMIN_QOS)
287   public RegionServerReportResponse regionServerReport(
288       RpcController controller, RegionServerReportRequest request) throws ServiceException {
289     try {
290       master.checkServiceStarted();
291       ClusterStatusProtos.ServerLoad sl = request.getLoad();
292       ServerName serverName = ProtobufUtil.toServerName(request.getServer());
293       ServerLoad oldLoad = master.serverManager.getLoad(serverName);
294       master.serverManager.regionServerReport(serverName, new ServerLoad(sl));
295       if (sl != null && master.metricsMaster != null) {
296         // Up our metrics.
297         master.metricsMaster.incrementRequests(sl.getTotalNumberOfRequests()
298           - (oldLoad != null ? oldLoad.getTotalNumberOfRequests() : 0));
299       }
300     } catch (IOException ioe) {
301       throw new ServiceException(ioe);
302     }
303     return RegionServerReportResponse.newBuilder().build();
304   }

308   public RegionServerStartupResponse regionServerStartup(
309       RpcController controller, RegionServerStartupRequest request) throws ServiceException {
310     // Register with server manager
311     try {
312       master.checkServiceStarted();
313       InetAddress ia = master.getRemoteInetAddress(
314         request.getPort(), request.getServerStartCode());
315       // if regionserver passed hostname to use,
316       // then use it instead of doing a reverse DNS lookup
317       ServerName rs = master.serverManager.regionServerStartup(request, ia);
318 
319       // Send back some config info
320       RegionServerStartupResponse.Builder resp = createConfigurationSubset();
321       NameStringPair.Builder entry = NameStringPair.newBuilder()
322         .setName(HConstants.KEY_FOR_HOSTNAME_SEEN_BY_MASTER)
323         .setValue(rs.getHostname());
324       resp.addMapEntries(entry.build());
325 
326       return resp.build();
327     } catch (IOException ioe) {
328       throw new ServiceException(ioe);
329     }
330   }

357   @Override
358   public AssignRegionResponse assignRegion(RpcController controller,
359       AssignRegionRequest req) throws ServiceException {
360     try {
361       final byte [] regionName = req.getRegion().getValue().toByteArray();
362       RegionSpecifierType type = req.getRegion().getType();
363       AssignRegionResponse arr = AssignRegionResponse.newBuilder().build();
364 
365       master.checkInitialized();
366       if (type != RegionSpecifierType.REGION_NAME) {
367         LOG.warn("assignRegion specifier type: expected: " + RegionSpecifierType.REGION_NAME
368           + " actual: " + type);
369       }
370       RegionStates regionStates = master.assignmentManager.getRegionStates();
371       HRegionInfo regionInfo = regionStates.getRegionInfo(regionName);
372       if (regionInfo == null) throw new UnknownRegionException(Bytes.toString(regionName));
373       if (master.cpHost != null) {
374         if (master.cpHost.preAssign(regionInfo)) {
375           return arr;
376         }
377       }
378       LOG.info(master.getClientIdAuditPrefix()
379         + " assign " + regionInfo.getRegionNameAsString());
380       master.assignmentManager.assign(regionInfo, true, true);
381       if (master.cpHost != null) {
382         master.cpHost.postAssign(regionInfo);
383       }
384       return arr;
385     } catch (IOException ioe) {
386       throw new ServiceException(ioe);
387     }
388   }

531       .toByteArray();
532     final boolean forcible = request.getForcible();
533     if (request.getRegionA().getType() != RegionSpecifierType.ENCODED_REGION_NAME
534         || request.getRegionB().getType() != RegionSpecifierType.ENCODED_REGION_NAME) {
535       LOG.warn("mergeRegions specifier type: expected: "
536         + RegionSpecifierType.ENCODED_REGION_NAME + " actual: region_a="
537         + request.getRegionA().getType() + ", region_b="
538         + request.getRegionB().getType());
539     }
540     RegionStates regionStates = master.assignmentManager.getRegionStates();
541     RegionState regionStateA = regionStates.getRegionState(Bytes.toString(encodedNameOfRegionA));
542     RegionState regionStateB = regionStates.getRegionState(Bytes.toString(encodedNameOfRegionB));
543     if (regionStateA == null || regionStateB == null) {
544       throw new ServiceException(new UnknownRegionException(
545           Bytes.toStringBinary(regionStateA == null ? encodedNameOfRegionA
546               : encodedNameOfRegionB)));
547     }
548 
549     if (!regionStateA.isOpened() || !regionStateB.isOpened()) {
550       throw new ServiceException(new MergeRegionException(
551         "Unable to merge regions not online " + regionStateA + ", " + regionStateB));
552     }
553 
554     HRegionInfo regionInfoA = regionStateA.getRegion();
555     HRegionInfo regionInfoB = regionStateB.getRegion();
556     if (regionInfoA.getReplicaId() != HRegionInfo.DEFAULT_REPLICA_ID ||
557         regionInfoB.getReplicaId() != HRegionInfo.DEFAULT_REPLICA_ID) {
558       throw new ServiceException(new MergeRegionException("Can't merge non-default replicas"));
559     }
560     if (regionInfoA.compareTo(regionInfoB) == 0) {
561       throw new ServiceException(new MergeRegionException(
562         "Unable to merge a region to itself " + regionInfoA + ", " + regionInfoB));
563     }
564 
565     if (!forcible && !HRegionInfo.areAdjacent(regionInfoA, regionInfoB)) {
566       throw new ServiceException(new MergeRegionException(
567         "Unable to merge not adjacent regions "
568           + regionInfoA.getRegionNameAsString() + ", "
569           + regionInfoB.getRegionNameAsString()
570           + " where forcible = " + forcible));
571     }
572 
573     try {
574       master.dispatchMergingRegions(regionInfoA, regionInfoB, forcible);
575     } catch (IOException ioe) {
576       throw new ServiceException(ioe);
577     }
578 
579     return DispatchMergingRegionsResponse.newBuilder().build();
580   }

617       String serviceName = call.getServiceName();

634           .mergeFrom(call.getRequest()).build();
635       final Message.Builder responseBuilder =
636           service.getResponsePrototype(methodDesc).newBuilderForType();
637       service.callMethod(methodDesc, execController, execRequest, new RpcCallback<Message>() {
638         @Override
639         public void run(Message message) {
640           if (message != null) {
641             responseBuilder.mergeFrom(message);
642           }
643         }
644       });
645       Message execResult = responseBuilder.build();
646 
647       if (execController.getFailedOn() != null) {
648         throw execController.getFailedOn();

651         ClientProtos.CoprocessorServiceResponse.newBuilder();

653         RegionSpecifierType.REGION_NAME, HConstants.EMPTY_BYTE_ARRAY));

1177       TableName dstTable = TableName.valueOf(request.getSnapshot().getTable());
1178       master.getNamespaceDescriptor(dstTable.getNamespaceAsString());

1255   @Override
1256   public UnassignRegionResponse unassignRegion(RpcController controller,
1257       UnassignRegionRequest req) throws ServiceException {
1258     try {
1259       final byte [] regionName = req.getRegion().getValue().toByteArray();
1260       RegionSpecifierType type = req.getRegion().getType();
1261       final boolean force = req.getForce();
1262       UnassignRegionResponse urr = UnassignRegionResponse.newBuilder().build();
1263 
1264       master.checkInitialized();
1265       if (type != RegionSpecifierType.REGION_NAME) {
1266         LOG.warn("unassignRegion specifier type: expected: " + RegionSpecifierType.REGION_NAME
1267           + " actual: " + type);
1268       }
1269       Pair<HRegionInfo, ServerName> pair =
1270         MetaTableAccessor.getRegion(master.getConnection(), regionName);
1271       if (pair == null) throw new UnknownRegionException(Bytes.toString(regionName));
1272       HRegionInfo hri = pair.getFirst();
1273       if (master.cpHost != null) {
1274         if (master.cpHost.preUnassign(hri, force)) {
1275           return urr;
1276         }
1277       }
1278       LOG.debug(master.getClientIdAuditPrefix() + " unassign " + hri.getRegionNameAsString()
1279           + " in current location if it is online and reassign.force=" + force);
1280       master.assignmentManager.unassign(hri, force);
1281       if (master.assignmentManager.getRegionStates().isRegionOffline(hri)) {
1282         LOG.debug("Region " + hri.getRegionNameAsString()
1283             + " is not online on any region server, reassigning it.");
1284         master.assignRegion(hri);
1285       }
1286       if (master.cpHost != null) {
1287         master.cpHost.postUnassign(hri, force);
1288       }
1289 
1290       return urr;
1291     } catch (IOException ioe) {
1292       throw new ServiceException(ioe);
1293     }
1294   }

1298   public ReportRegionStateTransitionResponse reportRegionStateTransition(RpcController c,
1299       ReportRegionStateTransitionRequest req) throws ServiceException {
1300     try {
1301       master.checkServiceStarted();
1302       RegionStateTransition rt = req.getTransition(0);
1303       TableName tableName = ProtobufUtil.toTableName(
1304         rt.getRegionInfo(0).getTableName());
1305       RegionStates regionStates = master.assignmentManager.getRegionStates();
1306       if (!(TableName.META_TABLE_NAME.equals(tableName)
1307           && regionStates.getRegionState(HRegionInfo.FIRST_META_REGIONINFO) != null)
1308             && !master.assignmentManager.isFailoverCleanupDone()) {
1309         // Meta region is assigned before master finishes the
1310         // failover cleanup. So no need this check for it
1311         throw new PleaseHoldException("Master is rebuilding user regions");
1312       }
1313       ServerName sn = ProtobufUtil.toServerName(req.getServer());
1314       String error = master.assignmentManager.onRegionTransition(sn, rt);
1315       ReportRegionStateTransitionResponse.Builder rrtr =
1316         ReportRegionStateTransitionResponse.newBuilder();
1317       if (error != null) {
1318         rrtr.setErrorMessage(error);
1319       }
1320       return rrtr.build();
1321     } catch (IOException ioe) {
1322       throw new ServiceException(ioe);
1323     }
