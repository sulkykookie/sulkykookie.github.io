24  package org.hibernate.dialect;
25  import java.sql.Types;
26  
27  import org.hibernate.cfg.Environment;
28  import org.hibernate.dialect.function.NoArgSQLFunction;
29  import org.hibernate.dialect.function.VarArgsSQLFunction;
30  import org.hibernate.type.StandardBasicTypes;
31  
32  /**
33   * An SQL dialect for Interbase.
34   *
35   * @author Gavin King
36   */
37  public class InterbaseDialect extends Dialect {
38  
39  	public InterbaseDialect() {
40  		super();
41  		registerColumnType( Types.BIT, "smallint" );
42  		registerColumnType( Types.BIGINT, "numeric(18,0)" );
43  		registerColumnType( Types.SMALLINT, "smallint" );
44  		registerColumnType( Types.TINYINT, "smallint" );
45  		registerColumnType( Types.INTEGER, "integer" );
46  		registerColumnType( Types.CHAR, "char(1)" );
47  		registerColumnType( Types.VARCHAR, "varchar($l)" );
48  		registerColumnType( Types.FLOAT, "float" );
49  		registerColumnType( Types.DOUBLE, "double precision" );
50  		registerColumnType( Types.DATE, "date" );
51  		registerColumnType( Types.TIME, "time" );
52  		registerColumnType( Types.TIMESTAMP, "timestamp" );
53  		registerColumnType( Types.VARBINARY, "blob" );
54  		registerColumnType( Types.NUMERIC, "numeric($p,$s)" );
55  		registerColumnType( Types.BLOB, "blob" );
56  		registerColumnType( Types.CLOB, "blob sub_type 1" );
57  		registerColumnType( Types.BOOLEAN, "smallint" );
58  		
59  		registerFunction( "concat", new VarArgsSQLFunction( StandardBasicTypes.STRING, "(","||",")" ) );
60  		registerFunction("current_date", new NoArgSQLFunction("current_date", StandardBasicTypes.DATE, false) );
61  
62  		getDefaultProperties().setProperty(Environment.STATEMENT_BATCH_SIZE, NO_BATCH);
63  	}
64  
65  	public String getAddColumnString() {
66  		return "add";
67  	}
68  
69  	public String getSequenceNextValString(String sequenceName) {
70  		return "select " + getSelectSequenceNextValString( sequenceName ) + " from RDB$DATABASE";
71  	}
72  
73  	public String getSelectSequenceNextValString(String sequenceName) {
74  		return "gen_id( " + sequenceName + ", 1 )";
75  	}
76  
77  	public String getCreateSequenceString(String sequenceName) {
78  		return "create generator " + sequenceName;
79  	}
80  
81  	public String getDropSequenceString(String sequenceName) {
82  		return "delete from RDB$GENERATORS where RDB$GENERATOR_NAME = '" + sequenceName.toUpperCase() + "'";
83  	}
84  
85  	public String getQuerySequencesString() {
86  		return "select RDB$GENERATOR_NAME from RDB$GENERATORS";
87  	}
88  	
89  	public String getForUpdateString() {
90  		return " with lock";
91  	}
92  	public String getForUpdateString(String aliases) {
93  		return " for update of " + aliases + " with lock";
94  	}
95  
96  	public boolean supportsSequences() {
97  		return true;
98  	}
99  
100 	public boolean supportsLimit() {
101 		return true;
102 	}
103 
104 	public String getLimitString(String sql, boolean hasOffset) {
105 		return hasOffset ? sql + " rows ? to ?" : sql + " rows ?";
106 	}
107 
108 	public boolean bindLimitParametersFirst() {
109 		return false;
110 	}
111 
112 	public boolean bindLimitParametersInReverseOrder() {
113 		return false;
114 	}
115 
116 	public String getCurrentTimestampCallString() {
117 		// TODO : not sure which (either?) is correct, could not find docs on how to do this.
118 		// did find various blogs and forums mentioning that select CURRENT_TIMESTAMP
119 		// does not work...
120 		return "{?= call CURRENT_TIMESTAMP }";
121 //		return "select CURRENT_TIMESTAMP from RDB$DATABASE";
122 	}
123 
124 	public boolean isCurrentTimestampSelectStringCallable() {
125 		return true;
126 	}
127 }
