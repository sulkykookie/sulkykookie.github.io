1   package org.hibernate.envers.strategy;
2   
3   import java.io.Serializable;
4   
5   import org.hibernate.Session;
6   import org.hibernate.collection.spi.PersistentCollection;
7   import org.hibernate.envers.configuration.AuditConfiguration;
8   import org.hibernate.envers.configuration.GlobalConfiguration;
9   import org.hibernate.envers.entities.mapper.PersistentCollectionChangeData;
10  import org.hibernate.envers.entities.mapper.relation.MiddleComponentData;
11  import org.hibernate.envers.entities.mapper.relation.MiddleIdData;
12  import org.hibernate.envers.tools.query.Parameters;
13  import org.hibernate.envers.tools.query.QueryBuilder;
14  
15  /**
16   * Behaviours of different audit strategy for populating audit data.
17   * 
18   * @author Stephanie Pau
19   * @author Adam Warski (adam at warski dot org)
20   */
21  public interface AuditStrategy {
22      /**
23       * Perform the persistence of audited data for regular entities.
24       * 
25       * @param session Session, which can be used to persist the data.
26       * @param entityName Name of the entity, in which the audited change happens
27       * @param auditCfg Audit configuration
28       * @param id Id of the entity.
29       * @param data Audit data to persist
30       * @param revision Current revision data
31       */
32      void perform(Session session, String entityName, AuditConfiguration auditCfg, Serializable id, Object data,
33                   Object revision);
34  
35      /**
36       * Perform the persistence of audited data for collection ("middle") entities.
37       *
38       * @param session Session, which can be used to persist the data.
39  	 * @param entityName Name of the entity, in which the audited change happens.
40  	 * @param propertyName The name of the property holding the {@link PersistentCollection}.
41       * @param auditCfg Audit configuration
42       * @param persistentCollectionChangeData Collection change data to be persisted.
43       * @param revision Current revision data
44       */
45      void performCollectionChange(Session session, String entityName, String propertyName, AuditConfiguration auditCfg,
46                                   PersistentCollectionChangeData persistentCollectionChangeData, Object revision);
47      
48  
49      /**
50  	 * Update the rootQueryBuilder with an extra WHERE clause to restrict the revision for a two-entity relation.
51  	 * This WHERE clause depends on the AuditStrategy, as follows:
52  	 * <ul>
53  	 * <li>For {@link DefaultAuditStrategy} a subquery is created: 
54  	 * <p><code>e.revision = (SELECT max(...) ...)</code></p>
55  	 * </li>
56  	 * <li>for {@link ValidityAuditStrategy} the revision-end column is used: 
57  	 * <p><code>e.revision <= :revision and (e.endRevision > :revision or e.endRevision is null)</code></p>
58  	 * </li>
59  	 * </ul>
60  	 * 
61  	 * @param globalCfg the {@link GlobalConfiguration}
62       * @param rootQueryBuilder the {@link QueryBuilder} that will be updated
63  	 * @param parameters root parameters to which restrictions shall be added
64       * @param revisionProperty property of the revision column
65       * @param revisionEndProperty property of the revisionEnd column (only used for {@link ValidityAuditStrategy})
66       * @param addAlias {@code boolean} indicator if a left alias is needed
67       * @param idData id-information for the two-entity relation (only used for {@link DefaultAuditStrategy})
68       * @param revisionPropertyPath path of the revision property (only used for {@link ValidityAuditStrategy})
69       * @param originalIdPropertyName name of the id property (only used for {@link ValidityAuditStrategy})
70       * @param alias1 an alias used for subquery (only used for {@link ValidityAuditStrategy})
71       * @param alias2 an alias used for subquery (only used for {@link ValidityAuditStrategy})
72  	 * @param inclusive indicates whether revision number shall be treated as inclusive or exclusive
73       */
74  	void addEntityAtRevisionRestriction(GlobalConfiguration globalCfg, QueryBuilder rootQueryBuilder, Parameters parameters,
75  			String revisionProperty, String revisionEndProperty, boolean addAlias, MiddleIdData idData, 
76  			String revisionPropertyPath, String originalIdPropertyName, String alias1, String alias2, boolean inclusive);
77  
78  	/**
79  	 * Update the rootQueryBuilder with an extra WHERE clause to restrict the revision for a middle-entity 
80  	 * association. This WHERE clause depends on the AuditStrategy, as follows:
81  	 * <ul>
82  	 * <li>For {@link DefaultAuditStrategy} a subquery is created: 
83  	 * <p><code>e.revision = (SELECT max(...) ...)</code></p>
84  	 * </li>
85  	 * <li>for {@link ValidityAuditStrategy} the revision-end column is used: 
86  	 * <p><code>e.revision <= :revision and (e.endRevision > :revision or e.endRevision is null)</code></p>
87  	 * </li>
88  	 * </ul>
89  	 * 
90  	 * @param rootQueryBuilder the {@link QueryBuilder} that will be updated
91  	 * @param parameters root parameters to which restrictions shall be added
92       * @param revisionProperty property of the revision column
93       * @param revisionEndProperty property of the revisionEnd column (only used for {@link ValidityAuditStrategy})
94       * @param addAlias {@code boolean} indicator if a left alias is needed
95       * @param referencingIdData id-information for the middle-entity association (only used for {@link DefaultAuditStrategy})
96  	 * @param versionsMiddleEntityName name of the middle-entity
97  	 * @param eeOriginalIdPropertyPath name of the id property (only used for {@link ValidityAuditStrategy})
98  	 * @param revisionPropertyPath path of the revision property (only used for {@link ValidityAuditStrategy})
99  	 * @param originalIdPropertyName name of the id property (only used for {@link ValidityAuditStrategy})
100 	 * @param alias1 an alias used for subqueries (only used for {@link DefaultAuditStrategy})
101 	 * @param inclusive indicates whether revision number shall be treated as inclusive or exclusive
102 	 * @param componentDatas information about the middle-entity relation
103 	 */
104 	void addAssociationAtRevisionRestriction(QueryBuilder rootQueryBuilder, Parameters parameters, String revisionProperty,
105 			String revisionEndProperty, boolean addAlias, MiddleIdData referencingIdData, 
106 			String versionsMiddleEntityName, String eeOriginalIdPropertyPath, String revisionPropertyPath,
107 			String originalIdPropertyName, String alias1, boolean inclusive, MiddleComponentData... componentDatas);
108 }
