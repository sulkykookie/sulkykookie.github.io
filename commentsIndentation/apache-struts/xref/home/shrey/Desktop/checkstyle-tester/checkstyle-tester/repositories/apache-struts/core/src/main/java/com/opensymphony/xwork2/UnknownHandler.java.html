19  package com.opensymphony.xwork2;
20  
21  import com.opensymphony.xwork2.config.entities.ActionConfig;
22  import org.apache.struts2.StrutsException;
23  
24  /**
25   * Handles cases when the result or action is unknown.
26   *
27   * <p>
28   * This allows other classes like Struts plugins to provide intelligent defaults easier.
29   * </p>
30   */
31  public interface UnknownHandler {
32      
33      /**
34       * Handles the case when an action configuration is unknown.  Implementations can return a new ActionConfig
35       * to be used to process the request.
36       * 
37       * @param namespace The namespace
38       * @param actionName The action name
39       * @return An generated ActionConfig, can return <tt>null</tt>
40       * @throws StrutsException in case of errors
41       */
42      ActionConfig handleUnknownAction(String namespace, String actionName) throws StrutsException;
43      
44      /**
45       * Handles the case when a result cannot be found for an action and result code. 
46       * 
47       * @param actionContext The action context
48       * @param actionName The action name
49       * @param actionConfig The action config
50       * @param resultCode The returned result code
51       * @return A result to be executed, can return <tt>null</tt>
52       * @throws StrutsException in case of errors
53       */
54      Result handleUnknownResult(ActionContext actionContext, String actionName, ActionConfig actionConfig, String resultCode) throws StrutsException;
55      
56      /**
57       * Handles the case when an action method cannot be found.  This method is responsible both for finding the method and executing it.
58       * 
59       * @since 2.1
60       * @param action The action object
61       * @param methodName The method name to call
62       * @return The result returned from invoking the action method, can return <tt>null</tt> if this instance cannot
63       *         handle such methodName to allow other handlers give a chance
64       */
65  	Object handleUnknownActionMethod(Object action, String methodName);
66  
67  }
