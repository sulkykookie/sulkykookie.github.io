43  /**
44   *
45   * Extracts the latest edition of the source code from a PVCS repository.
46   * PVCS is a version control system
47   * developed by <a href="http://www.merant.com/products/pvcs">Merant</a>.
48   * <br>
49   * Before using this tag, the user running ant must have access to the commands
50   * of PVCS (get and pcli) and must have access to the repository. Note that the way to specify
51   * the repository is platform dependent so use property to specify location of repository.
52   * <br>
53   * This version has been tested agains PVCS version 6.5 and 6.6 under Windows and Solaris.
54  
55   *
56   * <b>19-04-2001</b> <p>The task now has a more robust
57   * parser. It allows for platform independent file paths
58   * and supports file names with <i>()</i>. Thanks to Erik Husby for
59   * bringing the bug to my attention.
60   *
61   * <b>27-04-2001</b> <p>UNC paths are now handled properly.
62   * Fix provided by Don Jeffery. He also added an <i>UpdateOnly</i> flag
63   * that, when true, conditions the PVCS get using the -U option to only
64   * update those files that have a modification time (in PVCS) that is newer
65   * than the existing workfile.
66   *
67   * <b>25-10-2002</b> <p>Added a revision attribute that currently is a
68   * synonym for label, but in a future release the behavior of the label
69   * attribute will change to use the -v option of GET.  See bug #13847 for
70   * discussion.
71   *
72   */

111     /**
112      * Run the command.
113      * @param cmd the command line to use.
114      * @param out the output stream handler to use.
115      * @return the exit code of the command.
116      */
117     protected int runCmd(Commandline cmd, ExecuteStreamHandler out) {
118         try {
119             Project aProj = getProject();
120             Execute exe = new Execute(out);
121             exe.setAntRun(aProj);
122             exe.setWorkingDirectory(aProj.getBaseDir());
123             exe.setCommandline(cmd.getCommandline());
124             return exe.execute();
125         } catch (java.io.IOException e) {
126             String msg = "Failed executing: " + cmd.toString()
127                 + ". Exception: " + e.getMessage();
128             throw new BuildException(msg, getLocation());
129         }
130     }

132     private String getExecutable(String exe) {
133         StringBuffer correctedExe = new StringBuffer();
134         if (getPvcsbin() != null) {
135             if (pvcsbin.endsWith(File.separator)) {
136                 correctedExe.append(pvcsbin);
137             } else {
138                 correctedExe.append(pvcsbin).append(File.separator);
139             }
140         }
141         return correctedExe.append(exe).toString();
142     }

144     /**
145      * @exception org.apache.tools.ant.BuildException Something is stopping the build...
146      */
147     public void execute() throws org.apache.tools.ant.BuildException {
148         int result = 0;
149 
150         if (repository == null || repository.trim().equals("")) {
151             throw new BuildException("Required argument repository not specified");
152         }
153 
154         // Check workspace exists
155         // Launch PCLI listversionedfiles -z -aw
156         // Capture output
157         // build the command line from what we got the format is
158         Commandline commandLine = new Commandline();
159         commandLine.setExecutable(getExecutable(PCLI_EXE));
160 
161         commandLine.createArgument().setValue("lvf");
162         commandLine.createArgument().setValue("-z");
163         commandLine.createArgument().setValue("-aw");
164         if (getWorkspace() != null) {
165             commandLine.createArgument().setValue("-sp" + getWorkspace());
166         }
167         commandLine.createArgument().setValue("-pr" + getRepository());
168 
169         String uid = getUserId();
170 
171         if (uid != null) {
172             commandLine.createArgument().setValue("-id" + uid);
173         }
174 
175         // default pvcs project is "/"
176         if (getPvcsproject() == null && getPvcsprojects().isEmpty()) {
177             pvcsProject = "/";
178         }
179 
180         if (getPvcsproject() != null) {
181             commandLine.createArgument().setValue(getPvcsproject());
182         }
183         if (!getPvcsprojects().isEmpty()) {
184             Enumeration e = getPvcsprojects().elements();
185             while (e.hasMoreElements()) {
186                 String projectName = ((PvcsProject) e.nextElement()).getName();
187                 if (projectName == null || (projectName.trim()).equals("")) {
188                     throw new BuildException("name is a required attribute "
189                         + "of pvcsproject");
190                 }
191                 commandLine.createArgument().setValue(projectName);
192             }
193         }
194 
195         File tmp = null;
196         File tmp2 = null;
197         try {
198             Random rand = new Random(System.currentTimeMillis());
199             tmp = new File("pvcs_ant_" + rand.nextLong() + ".log");
200             FileOutputStream fos = new FileOutputStream(tmp);
201             tmp2 = new File("pvcs_ant_" + rand.nextLong() + ".log");
202             log(commandLine.describeCommand(), Project.MSG_VERBOSE);
203             try {
204                 result = runCmd(commandLine,
205                                 new PumpStreamHandler(fos,
206                                     new LogOutputStream(this,
207                                                         Project.MSG_WARN)));
208             } finally {
209                 FileUtils.close(fos);
210             }
211 
212             if (Execute.isFailure(result) && !ignorerc) {
213                 String msg = "Failed executing: " + commandLine.toString();
214                 throw new BuildException(msg, getLocation());
215             }
216 
217             if (!tmp.exists()) {
218                 throw new BuildException("Communication between ant and pvcs "
219                     + "failed. No output generated from executing PVCS "
220                     + "commandline interface \"pcli\" and \"get\"");
221             }
222 
223             // Create folders in workspace
224             log("Creating folders", Project.MSG_INFO);
225             createFolders(tmp);
226 
227             // Massage PCLI lvf output transforming '\' to '/' so get command works appropriately
228             massagePCLI(tmp, tmp2);
229 
230             // Launch get on output captured from PCLI lvf
231             commandLine.clearArgs();
232             commandLine.setExecutable(getExecutable(GET_EXE));
233 
234             if (getConfig() != null && getConfig().length() > 0) {
235                 commandLine.createArgument().setValue("-c" + getConfig());
236             }
237 
238             if (getForce() != null && getForce().equals("yes")) {
239                 commandLine.createArgument().setValue("-Y");
240             } else {
241                 commandLine.createArgument().setValue("-N");
242             }
243 
244             if (getPromotiongroup() != null) {
245                 commandLine.createArgument().setValue("-G"
246                     + getPromotiongroup());
247             } else {
248                 if (getLabel() != null) {
249                     commandLine.createArgument().setValue("-v" + getLabel());
250                 } else {
251                     if (getRevision() != null) {
252                         commandLine.createArgument().setValue("-r"
253                             + getRevision());
254                     }
255                 }
256             }
257 
258             if (updateOnly) {
259                 commandLine.createArgument().setValue("-U");
260             }
261 
262             commandLine.createArgument().setValue("@" + tmp2.getAbsolutePath());
263             log("Getting files", Project.MSG_INFO);
264             log("Executing " + commandLine.toString(), Project.MSG_VERBOSE);
265             result = runCmd(commandLine,
266                 new LogStreamHandler(this, Project.MSG_INFO, Project.MSG_WARN));
267             if (result != 0 && !ignorerc) {
268                 String msg = "Failed executing: " + commandLine.toString()
269                     + ". Return code was " + result;
270                 throw new BuildException(msg, getLocation());
271             }
272 
273         } catch (FileNotFoundException e) {
274             String msg = "Failed executing: " + commandLine.toString()
275                 + ". Exception: " + e.getMessage();
276             throw new BuildException(msg, getLocation());
277         } catch (IOException e) {
278             String msg = "Failed executing: " + commandLine.toString()
279                 + ". Exception: " + e.getMessage();
280             throw new BuildException(msg, getLocation());
281         } catch (ParseException e) {
282             String msg = "Failed executing: " + commandLine.toString()
283                 + ". Exception: " + e.getMessage();
284             throw new BuildException(msg, getLocation());
285         } finally {
286             if (tmp != null) {
287                 tmp.delete();
288             }
289             if (tmp2 != null) {
290                 tmp2.delete();
291             }
292         }
293     }

295     /**
296      * Parses the file and creates the folders specified in the output section
297      */
298     private void createFolders(File file) throws IOException, ParseException {
299         BufferedReader in = null;
300         try {
301             in = new BufferedReader(new FileReader(file));
302             MessageFormat mf = new MessageFormat(getFilenameFormat());
303             String line = in.readLine();
304             while (line != null) {
305                 log("Considering \"" + line + "\"", Project.MSG_VERBOSE);
306                 if (line.startsWith("\"\\")    // Checking for "\
307                     || line.startsWith("\"/")  // or           "/
308                                                // or           "X:\...
309                    || (line.length() > POS_3 && line.startsWith("\"")
310                         && Character.isLetter(line.charAt(POS_1))
311                         && String.valueOf(line.charAt(POS_2)).equals(":")
312                         && String.valueOf(line.charAt(POS_3)).equals("\\"))) {
313                     Object[] objs = mf.parse(line);
314                     String f = (String) objs[1];
315                     // Extract the name of the directory from the filename
316                     int index = f.lastIndexOf(File.separator);
317                     if (index > -1) {
318                         File dir = new File(f.substring(0, index));
319                         if (!dir.exists()) {
320                             log("Creating " + dir.getAbsolutePath(),
321                                 Project.MSG_VERBOSE);
322                             if (dir.mkdirs() || dir.isDirectory()) {
323                                 log("Created " + dir.getAbsolutePath(),
324                                     Project.MSG_INFO);
325                             } else {
326                                 log("Failed to create "
327                                     + dir.getAbsolutePath(),
328                                     Project.MSG_INFO);
329                             }
330                         } else {
331                             log(dir.getAbsolutePath() + " exists. Skipping",
332                                 Project.MSG_VERBOSE);
333                         }
334                     } else {
335                         log("File separator problem with " + line,
336                             Project.MSG_WARN);
337                     }
338                 } else {
339                     log("Skipped \"" + line + "\"", Project.MSG_VERBOSE);
340                 }
341                 line = in.readLine();
342             }
343         } finally {
344             FileUtils.close(in);
345         }
346     }

349     /**
350      * Simple hack to handle the PVCS command-line tools botch when
351      * handling UNC notation.
352      * @throws IOException if there is an error.
353      */
354     private void massagePCLI(File in, File out)
355         throws IOException {
356         BufferedReader inReader = null;
357         BufferedWriter outWriter = null;
358         try {
359             inReader = new BufferedReader(new FileReader(in));
360             outWriter = new BufferedWriter(new FileWriter(out));
361             String s = null;
362             while ((s = inReader.readLine()) != null) {
363                 String sNormal = s.replace('\\', '/');
364                 outWriter.write(sNormal);
365                 outWriter.newLine();
366             }
367         } finally {
368             FileUtils.close(inReader);
369             FileUtils.close(outWriter);
370         }
371     }
