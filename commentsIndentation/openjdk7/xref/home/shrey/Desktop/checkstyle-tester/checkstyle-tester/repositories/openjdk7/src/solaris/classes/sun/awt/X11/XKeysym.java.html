307         int  ndx = 0;
308         boolean getUppercase = false;
309         if ((ev.get_state() & XToolkit.numLockMask) != 0 &&
310              isKPEvent(ev)) {
311             keysym = getKeypadKeysym( ev );
312         } else {
313             // XXX: at this point, anything in keysym[23] is ignored.
314             //
315             // Shift & Lock are off ===> ndx = 0;
316             // Shift off & Lock on & Lock is CapsLock ===> ndx = 0;
317             //       if keysym[ndx] is lowecase alphabetic, then corresp. uppercase used.
318             // Shift on & Lock on & Lock is CapsLock ===> ndx == 1;
319             //       if keysym[ndx] is lowecase alphabetic, then corresp. uppercase used.
320             // Shift on || (Lock on & Lock is ShiftLock) ===> ndx = 1.
321             if ((ev.get_state() & XConstants.ShiftMask) == 0) {     // shift is off
322                 if ((ev.get_state() & XConstants.LockMask) == 0 ) {  // lock is off
323                     ndx = 0;
324                     getUppercase = false;
325                 } else if ((ev.get_state() & XConstants.LockMask) != 0 && // lock is on
326                      (XToolkit.modLockIsShiftLock == 0)) { // lock is capslock
327                     ndx = 0;
328                     getUppercase = true;
329                 } else if ((ev.get_state() & XConstants.LockMask) != 0 && // lock is on
330                      (XToolkit.modLockIsShiftLock != 0)) { // lock is shift lock
331                     ndx = 1;
332                     getUppercase = false;
333                 }
334             } else { // shift on
335                 if ((ev.get_state() & XConstants.LockMask) != 0 && // lock is on
336                      (XToolkit.modLockIsShiftLock == 0)) { // lock is capslock
337                     ndx = 1;
338                     getUppercase = true;
339                 } else {
340                     ndx = 1;
341                     getUppercase = false;
342                 }
343             }
344             keysym = xkeycode2keysym(ev, ndx);
345             if (getUppercase && (uppercaseKeysym =  getUppercaseAlphabetic( keysym )) != -1) {

347             }
348         }
349         return keysym;
