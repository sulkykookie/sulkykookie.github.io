136     @Test
137     public void testK1Example()
138         throws OrekitException, NoSuchFieldException, IllegalAccessException,
139                NoSuchMethodException, InvocationTargetException {
140         // the reference for this test is the example at the bottom of page 86, IERS conventions 2010 section 6.2.1
141         final PoissonSeriesParser<DerivativeStructure> k21Parser =
142                 new PoissonSeriesParser<DerivativeStructure>(18).
143                     withOptionalColumn(1).
144                     withDoodson(4, 3).
145                     withFirstDelaunay(10);
146         final String name = "/tides/tab6.5a-only-K1.txt";
147         final double pico = 1.0e-12;
148         final PoissonSeries<DerivativeStructure> c21Series =
149                 k21Parser.
150                 withSinCos(0, 17, pico, 18, pico).
151                 parse(getClass().getResourceAsStream(name), name);
152         final PoissonSeries<DerivativeStructure> s21Series =
153                 k21Parser.
154                 withSinCos(0, 18, -pico, 17, pico).
155                 parse(getClass().getResourceAsStream(name), name);
156         final UT1Scale ut1 = TimeScalesFactory.getUT1(IERSConventions.IERS_2010, false);
157         final TimeFunction<DerivativeStructure> gmstFunction = IERSConventions.IERS_2010.getGMSTFunction(ut1);
158         Method getNA = IERSConventions.class.getDeclaredMethod("getNutationArguments", TimeScale.class);
159         getNA.setAccessible(true);
160         final FundamentalNutationArguments arguments =
161                 (FundamentalNutationArguments) getNA.invoke(IERSConventions.IERS_2010, ut1);
162         TimeFunction<double[]> deltaCSFunction = new TimeFunction<double[]>() {
163             public double[] value(final AbsoluteDate date) {
164                 final BodiesElements elements = arguments.evaluateAll(date);
165                 return new double[] {
166                     0.0, c21Series.value(elements), s21Series.value(elements), 0.0, 0.0
167                 };
168             }
169         };
170 
171         SolidTidesField tf = new SolidTidesField(IERSConventions.IERS_2010.getLoveNumbers(),
172                                        deltaCSFunction,
173                                        IERSConventions.IERS_2010.getPermanentTide(),
174                                        IERSConventions.IERS_2010.getSolidPoleTide(ut1.getEOPHistory()),
175                                        FramesFactory.getITRF(IERSConventions.IERS_2010, false),
176                                        Constants.EIGEN5C_EARTH_EQUATORIAL_RADIUS,
177                                        Constants.EIGEN5C_EARTH_MU,
178                                        TideSystem.ZERO_TIDE,
179                                        CelestialBodyFactory.getSun(),
180                                        CelestialBodyFactory.getMoon());
181         Method frequencyDependentPart = SolidTidesField.class.getDeclaredMethod("frequencyDependentPart", AbsoluteDate.class, double[][].class, double[][].class);
182         frequencyDependentPart.setAccessible(true);
183         double[][] cachedCNM = new double[5][5];
184         double[][] cachedSNM = new double[5][5];
185 
186         AbsoluteDate t0 = new AbsoluteDate(2003, 5, 6, 13, 43, 32.125, TimeScalesFactory.getUTC());
187         for (double dt = 0; dt < Constants.JULIAN_DAY; dt += 300) {
188             AbsoluteDate date = t0.shiftedBy(dt);
189             for (int i = 0; i < cachedCNM.length; ++i) {
190                 Arrays.fill(cachedCNM[i], 0.0);
191                 Arrays.fill(cachedSNM[i], 0.0);
192             }
193             frequencyDependentPart.invoke(tf, date, cachedCNM, cachedSNM);
194             double thetaPlusPi = gmstFunction.value(date).getValue() + FastMath.PI;
195             Assert.assertEquals(470.9e-12 * FastMath.sin(thetaPlusPi) - 30.2e-12 * FastMath.cos(thetaPlusPi),
196                                 cachedCNM[2][1], 2.0e-25);
197             Assert.assertEquals(470.9e-12 * FastMath.cos(thetaPlusPi) + 30.2e-12 * FastMath.sin(thetaPlusPi),
198                                 cachedSNM[2][1], 2.0e-25);
199         }
200

245     @Test
246     public void testInterpolationAccuracy() throws OrekitException {
247 
248         // The shortest periods are slightly below one half day for the tidal waves
249         // considered here. This implies the sampling rate should be fast enough.
250         // The tuning parameters we have finally settled correspond to a two hours
251         // sample containing 12 points (i.e. one new point is computed every 10 minutes).
252         // The observed relative interpolation error with these settings are essentially
253         // due to Runge phenomenon at points sampling rate. Plotting the errors shows
254         // singular peaks pointing out of merely numerical noise.
255         final IERSConventions conventions = IERSConventions.IERS_2010;
256         Frame itrf    = FramesFactory.getITRF(conventions, true);
257         TimeScale utc = TimeScalesFactory.getUTC();
258         UT1Scale  ut1 = TimeScalesFactory.getUT1(conventions, true);
259         NormalizedSphericalHarmonicsProvider gravityField =
260                 GravityFieldFactory.getConstantNormalizedProvider(5, 5);
261 
262         SolidTidesField raw = new SolidTidesField(conventions.getLoveNumbers(),
263                                         conventions.getTideFrequencyDependenceFunction(ut1),
264                                         conventions.getPermanentTide(),
265                                         conventions.getSolidPoleTide(ut1.getEOPHistory()),
266                                         itrf, gravityField.getAe(), gravityField.getMu(),
267                                         gravityField.getTideSystem(),
268                                         CelestialBodyFactory.getSun(),
269                                         CelestialBodyFactory.getMoon());
270         int step     = 600;
271         int nbPoints = 12;
272         CachedNormalizedSphericalHarmonicsProvider interpolated =
273                 new CachedNormalizedSphericalHarmonicsProvider(raw, step, nbPoints,
274                                                                OrekitConfiguration.getCacheSlotsNumber(),
275                                                                7 * Constants.JULIAN_DAY,
276                                                                0.5 * Constants.JULIAN_DAY);
277 
278         // the following time range is located around the maximal observed error
279         AbsoluteDate start = new AbsoluteDate(2003, 6, 12, utc);
280         AbsoluteDate end   = start.shiftedBy(3 * Constants.JULIAN_DAY);
281         StreamingStatistics stat = new StreamingStatistics();
282         for (AbsoluteDate date = start; date.compareTo(end) < 0; date = date.shiftedBy(60)) {
283             NormalizedSphericalHarmonics rawHarmonics = raw.onDate(date);
284             NormalizedSphericalHarmonics interpolatedHarmonics = interpolated.onDate(date);
285 
286             for (int n = 2; n < 5; ++n) {
287                 for (int m = 0; m <= n; ++m) {
288 
289                     if (n < 4 || m < 3) {
290                         double cnmRaw    = rawHarmonics.getNormalizedCnm(n, m);
291                         double cnmInterp = interpolatedHarmonics.getNormalizedCnm(n, m);
292                         double errorC = (cnmInterp - cnmRaw) / FastMath.abs(cnmRaw);
293                         stat.addValue(errorC);
294 
295                         if (m > 0) {
296                             double snmRaw    = rawHarmonics.getNormalizedSnm(n, m);
297                             double snmInterp = interpolatedHarmonics.getNormalizedSnm(n, m);
298                             double errorS = (snmInterp - snmRaw) / FastMath.abs(snmRaw);
299                             stat.addValue(errorS);
300                         }
301                     }
302                 }
303             }
304         }
305         Assert.assertEquals(0.0, stat.getMean(), 2.0e-12);
306         Assert.assertTrue(stat.getStandardDeviation() < 2.0e-9);
307         Assert.assertTrue(stat.getMin() > -9.0e-8);
308         Assert.assertTrue(stat.getMax() <  2.2e-7);
309 
310     }
