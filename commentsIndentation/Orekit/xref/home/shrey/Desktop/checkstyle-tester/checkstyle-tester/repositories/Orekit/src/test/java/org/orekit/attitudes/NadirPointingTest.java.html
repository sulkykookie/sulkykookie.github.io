65      /** Test in the case of a spheric earth : nadir pointing shall be
66       * the same as earth center pointing
67       */
68      @Test
69      public void testSphericEarth() throws OrekitException {
70  
71          // Spheric earth shape
72          OneAxisEllipsoid earthShape = new OneAxisEllipsoid(6378136.460, 0., itrf);
73  
74          // Create nadir pointing attitude provider
75          NadirPointing nadirAttitudeLaw = new NadirPointing(FramesFactory.getEME2000(), earthShape);
76  
77          // Create earth center pointing attitude provider
78          BodyCenterPointing earthCenterAttitudeLaw =
79                  new BodyCenterPointing(FramesFactory.getEME2000(), earthShape);
80  
81          // Create satellite position as circular parameters
82          CircularOrbit circ =
83              new CircularOrbit(7178000.0, 0.5e-4, -0.5e-4, FastMath.toRadians(50.), FastMath.toRadians(270.),
84                                     FastMath.toRadians(5.300), PositionAngle.MEAN,
85                                     FramesFactory.getEME2000(), date, mu);
86  
87          // Get nadir attitude
88          Rotation rotNadir = nadirAttitudeLaw.getAttitude(circ, date, circ.getFrame()).getRotation();
89  
90          // Get earth center attitude
91          Rotation rotCenter = earthCenterAttitudeLaw.getAttitude(circ, date, circ.getFrame()).getRotation();
92  
93          // For a spheric earth, earth center pointing attitude and nadir pointing attitude
94          // shall be the same, i.e the composition of inverse earth pointing rotation
95          // with nadir pointing rotation shall be identity.
96          Rotation rotCompo = rotCenter.composeInverse(rotNadir, RotationConvention.VECTOR_OPERATOR);
97          double angle = rotCompo.getAngle();
98          Assert.assertEquals(angle, 0.0, Utils.epsilonAngle);
99  
100     }

102     /** Test in the case of an elliptic earth : nadir pointing shall be :
103      *   - the same as earth center pointing in case of equatorial or polar position
104      *   - different from earth center pointing in any other case
105      */
106     @Test
107     public void testNonSphericEarth() throws OrekitException {
108 
109         // Elliptic earth shape
110         OneAxisEllipsoid earthShape = new OneAxisEllipsoid(6378136.460, 1 / 298.257222101, itrf);
111 
112         // Create nadir pointing attitude provider
113         NadirPointing nadirAttitudeLaw = new NadirPointing(FramesFactory.getEME2000(), earthShape);
114 
115         // Create earth center pointing attitude provider
116         BodyCenterPointing earthCenterAttitudeLaw =
117                 new BodyCenterPointing(FramesFactory.getEME2000(), earthShape);
118 
119         //  Satellite on equatorial position
120         // **********************************
121         KeplerianOrbit kep =
122             new KeplerianOrbit(7178000.0, 1.e-8, FastMath.toRadians(50.), 0., 0.,
123                                     0., PositionAngle.TRUE, FramesFactory.getEME2000(), date, mu);
124 
125         // Get nadir attitude
126         Rotation rotNadir = nadirAttitudeLaw.getAttitude(kep, date, kep.getFrame()).getRotation();
127 
128         // Get earth center attitude
129         Rotation rotCenter = earthCenterAttitudeLaw.getAttitude(kep, date, kep.getFrame()).getRotation();
130 
131         // For a satellite at equatorial position, earth center pointing attitude and nadir pointing
132         // attitude shall be the same, i.e the composition of inverse earth pointing rotation
133         // with nadir pointing rotation shall be identity.
134         Rotation rotCompo = rotCenter.composeInverse(rotNadir, RotationConvention.VECTOR_OPERATOR);
135         double angle = rotCompo.getAngle();
136         Assert.assertEquals(0.0, angle, 5.e-6);
137 
138         //  Satellite on polar position
139         // *****************************
140         CircularOrbit circ =
141             new CircularOrbit(7178000.0, 1.e-5, 0., FastMath.toRadians(90.), 0.,
142                                    FastMath.toRadians(90.), PositionAngle.TRUE,
143                                    FramesFactory.getEME2000(), date, mu);
144 
145        // Get nadir attitude
146         rotNadir = nadirAttitudeLaw.getAttitude(circ, date, circ.getFrame()).getRotation();
147 
148         // Get earth center attitude
149         rotCenter = earthCenterAttitudeLaw.getAttitude(circ, date, circ.getFrame()).getRotation();
150 
151         // For a satellite at polar position, earth center pointing attitude and nadir pointing
152         // attitude shall be the same, i.e the composition of inverse earth pointing rotation
153         // with nadir pointing rotation shall be identity.
154         rotCompo = rotCenter.composeInverse(rotNadir, RotationConvention.VECTOR_OPERATOR);
155         angle = rotCompo.getAngle();
156         Assert.assertEquals(angle, 0.0, 5.e-6);
157 
158         //  Satellite on any position
159         // ***************************
160         circ =
161             new CircularOrbit(7178000.0, 1.e-5, 0., FastMath.toRadians(50.), 0.,
162                                    FastMath.toRadians(90.), PositionAngle.TRUE,
163                                    FramesFactory.getEME2000(), date, mu);
164 
165         // Get nadir attitude
166         rotNadir = nadirAttitudeLaw.getAttitude(circ, date, circ.getFrame()).getRotation();
167 
168         // Get earth center attitude
169         rotCenter = earthCenterAttitudeLaw.getAttitude(circ, date, circ.getFrame()).getRotation();
170 
171         // For a satellite at any position, earth center pointing attitude and nadir pointing
172         // and nadir pointing attitude shall not be the same, i.e the composition of inverse earth
173         // pointing rotation with nadir pointing rotation shall be different from identity.
174         rotCompo = rotCenter.composeInverse(rotNadir, RotationConvention.VECTOR_OPERATOR);
175         angle = rotCompo.getAngle();
176         Assert.assertEquals(angle, FastMath.toRadians(0.16797386586252272), Utils.epsilonAngle);
177     }

179     /** Vertical test : check that Z satellite axis is collinear to local vertical axis,
180         which direction is : (cos(lon)*cos(lat), sin(lon)*cos(lat), sin(lat)),
181         where lon et lat stand for observed point coordinates
182         (i.e satellite ones, since they are the same by construction,
183         but that's what is to test.
184      */
185     @Test
186     public void testVertical() throws OrekitException {
187 
188         // Elliptic earth shape
189         OneAxisEllipsoid earthShape = new OneAxisEllipsoid(6378136.460, 1 / 298.257222101, itrf);
190 
191         // Create earth center pointing attitude provider
192         NadirPointing nadirAttitudeLaw = new NadirPointing(FramesFactory.getEME2000(), earthShape);
193 
194         //  Satellite on any position
195         CircularOrbit circ =
196             new CircularOrbit(7178000.0, 1.e-5, 0., FastMath.toRadians(50.), 0.,
197                                    FastMath.toRadians(90.), PositionAngle.TRUE,
198                                    FramesFactory.getEME2000(), date, mu);
199 
200         //  Vertical test
201         // ***************
202         // Get observed ground point position/velocity
203         TimeStampedPVCoordinates pvTargetItrf = nadirAttitudeLaw.getTargetPV(circ, date, itrf);
204 
205         // Convert to geodetic coordinates
206         GeodeticPoint geoTarget = earthShape.transform(pvTargetItrf.getPosition(), itrf, date);
207 
208         // Compute local vertical axis
209         double xVert = FastMath.cos(geoTarget.getLongitude())*FastMath.cos(geoTarget.getLatitude());
210         double yVert = FastMath.sin(geoTarget.getLongitude())*FastMath.cos(geoTarget.getLatitude());
211         double zVert = FastMath.sin(geoTarget.getLatitude());
212         Vector3D targetVertical = new Vector3D(xVert, yVert, zVert);
213 
214         // Get attitude rotation state
215         Rotation rotSatEME2000 = nadirAttitudeLaw.getAttitude(circ, date, circ.getFrame()).getRotation();
216 
217         // Get satellite Z axis in EME2000 frame
218         Vector3D zSatEME2000 = rotSatEME2000.applyInverseTo(Vector3D.PLUS_K);
219         Vector3D zSatItrf = FramesFactory.getEME2000().getTransformTo(itrf, date).transformVector(zSatEME2000);
220 
221         // Check that satellite Z axis is collinear to local vertical axis
222         double angle= Vector3D.angle(zSatItrf, targetVertical);
223         Assert.assertEquals(0.0, FastMath.sin(angle), Utils.epsilonTest);
224 
225     }

227     /** Test the derivatives of the sliding target
228      */
229     @Test
230     public void testSlidingDerivatives() throws OrekitException {
231 
232         // Elliptic earth shape
233         OneAxisEllipsoid earthShape = new OneAxisEllipsoid(6378136.460, 1 / 298.257222101, itrf);
234 
235         // Create earth center pointing attitude provider
236         NadirPointing nadirAttitudeLaw = new NadirPointing(FramesFactory.getEME2000(), earthShape);
237 
238         //  Satellite on any position
239         CircularOrbit circ =
240             new CircularOrbit(7178000.0, 1.e-5, 0., FastMath.toRadians(50.), 0.,
241                                    FastMath.toRadians(90.), PositionAngle.TRUE,
242                                    FramesFactory.getEME2000(), date, mu);
243 
244         List<TimeStampedPVCoordinates> sample = new ArrayList<TimeStampedPVCoordinates>();
245         for (double dt = -0.1; dt < 0.1; dt += 0.05) {
246             Orbit o = circ.shiftedBy(dt);
247             sample.add(nadirAttitudeLaw.getTargetPV(o, o.getDate(), o.getFrame()));
248         }
249         TimeStampedPVCoordinates reference =
250                 TimeStampedPVCoordinates.interpolate(circ.getDate(),
251                                                      CartesianDerivativesFilter.USE_P, sample);
252 
253         TimeStampedPVCoordinates target =
254                 nadirAttitudeLaw.getTargetPV(circ, circ.getDate(), circ.getFrame());
255 
256         Assert.assertEquals(0.0,
257                             Vector3D.distance(reference.getPosition(),     target.getPosition()),
258                             1.0e-15 * reference.getPosition().getNorm());
259         Assert.assertEquals(0.0,
260                             Vector3D.distance(reference.getVelocity(),     target.getVelocity()),
261                             3.0e-11 * reference.getVelocity().getNorm());
262         Assert.assertEquals(0.0,
263                             Vector3D.distance(reference.getAcceleration(), target.getAcceleration()),
264                             1.0e-5 * reference.getAcceleration().getNorm());
265 
266     }

268     @Test
269     public void testSpin() throws OrekitException {
270 
271         // Elliptic earth shape
272         OneAxisEllipsoid earthShape = new OneAxisEllipsoid(6378136.460, 1 / 298.257222101, itrf);
273 
274         // Create earth center pointing attitude provider
275         NadirPointing law = new NadirPointing(FramesFactory.getEME2000(), earthShape);
276 
277         //  Satellite on any position
278         KeplerianOrbit orbit =
279             new KeplerianOrbit(7178000.0, 1.e-4, FastMath.toRadians(50.),
280                               FastMath.toRadians(10.), FastMath.toRadians(20.),
281                               FastMath.toRadians(30.), PositionAngle.MEAN,
282                               FramesFactory.getEME2000(), date, mu);
283 
284         Propagator propagator = new KeplerianPropagator(orbit, law, mu, 2500.0);
285 
286         double h = 0.1;
287         SpacecraftState sMinus = propagator.propagate(date.shiftedBy(-h));
288         SpacecraftState s0     = propagator.propagate(date);
289         SpacecraftState sPlus  = propagator.propagate(date.shiftedBy(h));
290 
291         // check spin is consistent with attitude evolution
292         double errorAngleMinus     = Rotation.distance(sMinus.shiftedBy(h).getAttitude().getRotation(),
293                                                        s0.getAttitude().getRotation());
294         double evolutionAngleMinus = Rotation.distance(sMinus.getAttitude().getRotation(),
295                                                        s0.getAttitude().getRotation());
296         Assert.assertEquals(0.0, errorAngleMinus, 4.0e-9 * evolutionAngleMinus);
297         double errorAnglePlus      = Rotation.distance(s0.getAttitude().getRotation(),
298                                                        sPlus.shiftedBy(-h).getAttitude().getRotation());
299         double evolutionAnglePlus  = Rotation.distance(s0.getAttitude().getRotation(),
300                                                        sPlus.getAttitude().getRotation());
301         Assert.assertEquals(0.0, errorAnglePlus, 8.0e-9 * evolutionAnglePlus);
302 
303         Vector3D spin0 = s0.getAttitude().getSpin();
304         Rotation rM = sMinus.getAttitude().getRotation();
305         Rotation rP = sPlus.getAttitude().getRotation();
306         Vector3D reference = AngularCoordinates.estimateRate(rM, rP, 2 * h);
307         Assert.assertTrue(Rotation.distance(rM, rP) > 2.0e-4);
308         Assert.assertEquals(0.0, spin0.subtract(reference).getNorm(), 2.0e-6);
309 
310     }
