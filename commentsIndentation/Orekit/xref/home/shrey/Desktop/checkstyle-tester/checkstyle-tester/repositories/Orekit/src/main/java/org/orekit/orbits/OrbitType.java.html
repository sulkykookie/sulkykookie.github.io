29  /** Enumerate for {@link Orbit orbital} parameters types.
30   */
31  public enum OrbitType {
32  
33      /** Type for propagation in {@link CartesianOrbit Cartesian parameters}. */
34      CARTESIAN {
35  
36          /** {@inheritDoc} */
37          public Orbit convertType(final Orbit orbit) {
38              return (orbit.getType() == this) ? orbit : new CartesianOrbit(orbit);
39          }
40  
41          /** {@inheritDoc} */
42          public void mapOrbitToArray(final Orbit orbit, final PositionAngle type,
43                                      final double[] stateVector) {
44  
45              final PVCoordinates pv = orbit.getPVCoordinates();
46              final Vector3D      p  = pv.getPosition();
47              final Vector3D      v  = pv.getVelocity();
48  
49              stateVector[0] = p.getX();
50              stateVector[1] = p.getY();
51              stateVector[2] = p.getZ();
52              stateVector[3] = v.getX();
53              stateVector[4] = v.getY();
54              stateVector[5] = v.getZ();
55  
56          }
57  
58          /** {@inheritDoc} */
59          public Orbit mapArrayToOrbit(final double[] stateVector, final PositionAngle type,
60                                       final AbsoluteDate date, final double mu, final Frame frame) {
61  
62              final Vector3D p     = new Vector3D(stateVector[0], stateVector[1], stateVector[2]);
63              final double r2      = p.getNormSq();
64              final Vector3D v     = new Vector3D(stateVector[3], stateVector[4], stateVector[5]);
65              final Vector3D a     = new Vector3D(-mu / (r2 * FastMath.sqrt(r2)), p);
66              return new CartesianOrbit(new PVCoordinates(p, v, a), frame, date, mu);
67  
68          }
69  
70          /** {@inheritDoc} */
71          public ParameterDriversList getDrivers(final double dP, final Orbit orbit, final PositionAngle type)
72              throws OrekitException {
73              final ParameterDriversList drivers = new ParameterDriversList();
74              final double[] array = new double[6];
75              mapOrbitToArray(orbit, type, array);
76              final double[] scale = scale(dP, orbit);
77              drivers.add(new ParameterDriver(POS_X, array[0], scale[0], Double.NEGATIVE_INFINITY, Double.POSITIVE_INFINITY));
78              drivers.add(new ParameterDriver(POS_Y, array[1], scale[1], Double.NEGATIVE_INFINITY, Double.POSITIVE_INFINITY));
79              drivers.add(new ParameterDriver(POS_Z, array[2], scale[2], Double.NEGATIVE_INFINITY, Double.POSITIVE_INFINITY));
80              drivers.add(new ParameterDriver(VEL_X, array[3], scale[3], Double.NEGATIVE_INFINITY, Double.POSITIVE_INFINITY));
81              drivers.add(new ParameterDriver(VEL_Y, array[4], scale[4], Double.NEGATIVE_INFINITY, Double.POSITIVE_INFINITY));
82              drivers.add(new ParameterDriver(VEL_Z, array[5], scale[5], Double.NEGATIVE_INFINITY, Double.POSITIVE_INFINITY));
83              return drivers;
84          }
85  
86      },

119         /** {@inheritDoc} */
120         public ParameterDriversList getDrivers(final double dP, final Orbit orbit, final PositionAngle type)
121             throws OrekitException {
122             final ParameterDriversList drivers = new ParameterDriversList();
123             final double[] array = new double[6];
124             mapOrbitToArray(orbit, type, array);
125             final double[] scale = scale(dP, orbit);
126             final String name = type == PositionAngle.MEAN ?
127                                     MEAN_LAT_ARG :
128                                     type == PositionAngle.ECCENTRIC ? ECC_LAT_ARG : TRUE_LAT_ARG;
129             drivers.add(new ParameterDriver(A,    array[0], scale[0],  0.0, Double.POSITIVE_INFINITY));
130             drivers.add(new ParameterDriver(E_X,  array[1], scale[1], -1.0, 1.0));
131             drivers.add(new ParameterDriver(E_Y,  array[2], scale[2], -1.0, 1.0));
132             drivers.add(new ParameterDriver(INC,  array[3], scale[3],  0.0, FastMath.PI));
133             drivers.add(new ParameterDriver(RAAN, array[4], scale[4], Double.NEGATIVE_INFINITY, Double.POSITIVE_INFINITY));
134             drivers.add(new ParameterDriver(name, array[5], scale[5], Double.NEGATIVE_INFINITY, Double.POSITIVE_INFINITY));
135             return drivers;
136         }

361     /** Get parameters drivers initialized from a reference orbit.
362      * @param dP user specified position error
363      * @param orbit reference orbit
364      * @param type type of the angle
365      * @return parameters drivers initialized from reference orbit
366      * @exception OrekitException if Jacobian is singular
367      */
368     public abstract ParameterDriversList getDrivers(final double dP, final Orbit orbit,
369                                                     final PositionAngle type)
370         throws OrekitException;

372     /** Compute scaling factor for parameters drivers.
373      * <p>
374      * The scales are estimated from partial derivatives properties of orbits,
375      * starting from a scalar position error specified by the user.
376      * Considering the energy conservation equation V = sqrt(mu (2/r - 1/a)),
377      * we get at constant energy (i.e. on a Keplerian trajectory):
378      * <pre>
379      * V² r |dV| = mu |dr|
380      * </pre>
381      * <p> So we deduce a scalar velocity error consistent with the position error.
382      * From here, we apply orbits Jacobians matrices to get consistent scales
383      * on orbital parameters.
384      *
385      * @param dP user specified position error
386      * @param orbit reference orbit
387      * @return scaling factor array
388      * @exception OrekitException if Jacobian is singular
389      */
390     protected double[] scale(final double dP, final Orbit orbit)
391         throws OrekitException {
392 
393         // estimate the scalar velocity error
394         final PVCoordinates pv = orbit.getPVCoordinates();
395         final double r2 = pv.getPosition().getNormSq();
396         final double v  = pv.getVelocity().getNorm();
397         final double dV = orbit.getMu() * dP / (v * r2);
398 
399         final double[] scale = new double[6];
400 
401         // convert the orbit to the desired type
402         final double[][] jacobian = new double[6][6];
403         final Orbit converted = convertType(orbit);
404         converted.getJacobianWrtCartesian(PositionAngle.TRUE, jacobian);
405 
406         for (int i = 0; i < 6; ++i) {
407             final double[] row = jacobian[i];
408             scale[i] = FastMath.abs(row[0]) * dP +
409                             FastMath.abs(row[1]) * dP +
410                             FastMath.abs(row[2]) * dP +
411                             FastMath.abs(row[3]) * dV +
412                             FastMath.abs(row[4]) * dV +
413                             FastMath.abs(row[5]) * dV;
414             if (Double.isNaN(scale[i])) {
415                 throw new OrekitException(OrekitMessages.SINGULAR_JACOBIAN_FOR_ORBIT_TYPE, this);
416             }
417         }
418 
419         return scale;
420 
421     }
