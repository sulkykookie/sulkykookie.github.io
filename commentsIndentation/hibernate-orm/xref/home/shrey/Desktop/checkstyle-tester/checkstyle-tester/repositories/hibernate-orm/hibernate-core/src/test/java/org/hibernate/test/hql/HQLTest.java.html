344 		assertTranslation("from Animal a where abs(a.bodyWeight-:param) < 2.0");
345 		assertTranslation("from Animal a where abs(:param - a.bodyWeight) < 2.0");
346 		assertTranslation("from Animal where abs(:x - :y) < 2.0");
347 		assertTranslation("from Animal where lower(upper(:foo)) like 'f%'");
348 		if ( ! ( getDialect() instanceof SybaseDialect ) &&  ! ( getDialect() instanceof Sybase11Dialect ) &&  ! ( getDialect() instanceof SybaseASE15Dialect ) && ! ( getDialect() instanceof SQLServerDialect ) ) {
349 			// Transact-SQL dialects (except SybaseAnywhereDialect) map the length function -> len;

351 			// SybaseAnywhereDialect supports the length function

( getDialect() instanceof Sybase11Dialect ) && !( getDialect() instanceof SybaseASE15Dialect ) && ! ( getDialect() instanceof SybaseAnywhereDialect ) && ! ( getDialect() instanceof SQLServerDialect ) ) {
