108 			// expected result
109 		}

261 		assertEquals( Type.PersistenceType.EMBEDDABLE, rooms.getElementType().getPersistenceType() );
262 
263 		final MapAttribute<House,String,Room> roomsByName = entityType.getDeclaredMap(

268 		assertEquals( Type.PersistenceType.BASIC, roomsByName.getKeyType().getPersistenceType() );
269 		assertEquals( PluralAttribute.CollectionType.MAP, roomsByName.getCollectionType() );

278 	public void testHierarchy() {
279 		final EntityType<Cat> cat = entityManagerFactory().getMetamodel().entity( Cat.class );
280 		assertNotNull( cat );
281 		assertEquals( 7, cat.getAttributes().size() );
282 		assertEquals( 1, cat.getDeclaredAttributes().size() );
283 		ensureProperMember(cat.getDeclaredAttributes());
284 
285 		assertTrue( cat.hasVersionAttribute() );
286 		assertEquals( "version", cat.getVersion(Long.class).getName() );
287 		verifyDeclaredVersionNotPresent( cat );
288 		verifyDeclaredIdNotPresentAndIdPresent(cat);
289 
290 		assertEquals( Type.PersistenceType.MAPPED_SUPERCLASS, cat.getSupertype().getPersistenceType() );
291 		MappedSuperclassType<Cattish> cattish = (MappedSuperclassType<Cattish>) cat.getSupertype();
292 		assertEquals( 6, cattish.getAttributes().size() );
293 		assertEquals( 1, cattish.getDeclaredAttributes().size() );
294 		ensureProperMember(cattish.getDeclaredAttributes());
295 
296 		assertTrue( cattish.hasVersionAttribute() );
297 		assertEquals( "version", cattish.getVersion(Long.class).getName() );
298 		verifyDeclaredVersionNotPresent( cattish );
299 		verifyDeclaredIdNotPresentAndIdPresent(cattish);
300 
301 		assertEquals( Type.PersistenceType.ENTITY, cattish.getSupertype().getPersistenceType() );
302 		EntityType<Feline> feline = (EntityType<Feline>) cattish.getSupertype();
303 		assertEquals( 5, feline.getAttributes().size() );
304 		assertEquals( 1, feline.getDeclaredAttributes().size() );
305 		ensureProperMember(feline.getDeclaredAttributes());
306 
307 		assertTrue( feline.hasVersionAttribute() );
308 		assertEquals( "version", feline.getVersion(Long.class).getName() );
309 		verifyDeclaredVersionNotPresent( feline );
310 		verifyDeclaredIdNotPresentAndIdPresent(feline);
311 
312 		assertEquals( Type.PersistenceType.MAPPED_SUPERCLASS, feline.getSupertype().getPersistenceType() );
313 		MappedSuperclassType<Animal> animal = (MappedSuperclassType<Animal>) feline.getSupertype();
314 		assertEquals( 4, animal.getAttributes().size() );
315 		assertEquals( 2, animal.getDeclaredAttributes().size() );
316 		ensureProperMember(animal.getDeclaredAttributes());
317 
318 		assertTrue( animal.hasVersionAttribute() );
319 		assertEquals( "version", animal.getVersion(Long.class).getName() );
320 		verifyDeclaredVersionNotPresent( animal );
321 		assertEquals( "id", animal.getId(Long.class).getName() );
322 		final SingularAttribute<Animal, Long> id = animal.getDeclaredId( Long.class );
323 		assertEquals( "id", id.getName() );
324 		assertNotNull( id.getJavaMember() );
325 
326 		assertEquals( Type.PersistenceType.MAPPED_SUPERCLASS, animal.getSupertype().getPersistenceType() );
327 		MappedSuperclassType<Thing> thing = (MappedSuperclassType<Thing>) animal.getSupertype();
328 		assertEquals( 2, thing.getAttributes().size() );
329 		assertEquals( 2, thing.getDeclaredAttributes().size() );
330 		ensureProperMember(thing.getDeclaredAttributes());
331 		final SingularAttribute<Thing, Double> weight = thing.getDeclaredSingularAttribute( "weight", Double.class );
332 		assertEquals( Double.class, weight.getJavaType() );
333 
334 		assertEquals( "version", thing.getVersion(Long.class).getName() );
335 		final SingularAttribute<Thing, Long> version = thing.getDeclaredVersion( Long.class );
336 		assertEquals( "version", version.getName() );
337 		assertNotNull( version.getJavaMember() );
338 		assertNull( thing.getId( Long.class ) );
339 
340 		assertNull( thing.getSupertype() );
341 	}

345 		final EntityType<Parent> parent = entityManagerFactory().getMetamodel().entity( Parent.class );

358 		assertNotNull(siblings);
359 		final SetAttribute<? super Parent.Relatives, ?> siblingsCollection = siblings.getSet( "siblings" );
360 		assertNotNull( siblingsCollection );
361 		final Type<?> collectionElement = siblingsCollection.getElementType();
362 		assertNotNull( collectionElement );
363 		assertEquals( collectionElement, child );
364 	}

366 	private void ensureProperMember(Set<?> attributes) {
367 		//we do not update the set so we are safe
368 		@SuppressWarnings( "unchecked" )
369 		final Set<Attribute<?, ?>> safeAttributes = ( Set<Attribute<?, ?>> ) attributes;
370 		for (Attribute<?,?> attribute : safeAttributes ) {
371 			final String name = attribute.getJavaMember().getName();
372 			assertNotNull( attribute.getJavaMember() );
373 			assertTrue( name.toLowerCase().endsWith( attribute.getName().toLowerCase() ) );
374 		}
375 	}
