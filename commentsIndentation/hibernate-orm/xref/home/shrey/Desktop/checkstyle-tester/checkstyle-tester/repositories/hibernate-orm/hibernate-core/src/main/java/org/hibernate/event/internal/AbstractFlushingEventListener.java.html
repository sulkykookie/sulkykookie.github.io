72  	/**
73  	 * Coordinates the processing necessary to get things ready for executions
74  	 * as db calls by preping the session caches and moving the appropriate
75  	 * entities and collections to their respective execution queues.
76  	 *
77  	 * @param event The flush event.
78  	 * @throws HibernateException Error flushing caches to execution queues.
79  	 */
80  	protected void flushEverythingToExecutions(FlushEvent event) throws HibernateException {
81  
82  		LOG.trace( "Flushing session" );
83  
84  		EventSource session = event.getSession();
85  
86  		final PersistenceContext persistenceContext = session.getPersistenceContext();
87  		session.getInterceptor().preFlush( new LazyIterator( persistenceContext.getEntitiesByKey() ) );
88  
89  		prepareEntityFlushes( session, persistenceContext );
90  		// we could move this inside if we wanted to
91  		// tolerate collection initializations during
92  		// collection dirty checking:
93  		prepareCollectionFlushes( persistenceContext );
94  		// now, any collections that are initialized
95  		// inside this block do not get updated - they
96  		// are ignored until the next flush
97  
98  		persistenceContext.setFlushing( true );
99  		try {
100 			int entityCount = flushEntities( event, persistenceContext );
101 			int collectionCount = flushCollections( session, persistenceContext );
102 
103 			event.setNumberOfEntitiesProcessed( entityCount );
104 			event.setNumberOfCollectionsProcessed( collectionCount );
105 		}
106 		finally {
107 			persistenceContext.setFlushing(false);
108 		}
109 
110 		//some statistics
111 		logFlushResults( event );
112 	}
113 
114 	@SuppressWarnings( value = {"unchecked"} )
115 	private void logFlushResults(FlushEvent event) {
116 		if ( !LOG.isDebugEnabled() ) {
117 			return;
118 		}
119 		final EventSource session = event.getSession();
120 		final PersistenceContext persistenceContext = session.getPersistenceContext();
121 		LOG.debugf(
122 				"Flushed: %s insertions, %s updates, %s deletions to %s objects",
123 				session.getActionQueue().numberOfInsertions(),
124 				session.getActionQueue().numberOfUpdates(),
125 				session.getActionQueue().numberOfDeletions(),
126 				persistenceContext.getNumberOfManagedEntities()
127 		);
128 		LOG.debugf(
129 				"Flushed: %s (re)creations, %s updates, %s removals to %s collections",
130 				session.getActionQueue().numberOfCollectionCreations(),
131 				session.getActionQueue().numberOfCollectionUpdates(),
132 				session.getActionQueue().numberOfCollectionRemovals(),
133 				persistenceContext.getCollectionEntries().size()
134 		);
135 		new EntityPrinter( session.getFactory() ).toString(
136 				persistenceContext.getEntitiesByKey().entrySet()
137 		);
138 	}

140 	/**
141 	 * process cascade save/update at the start of a flush to discover
142 	 * any newly referenced entity that must be passed to saveOrUpdate(),
143 	 * and also apply orphan delete
144 	 */
145 	private void prepareEntityFlushes(EventSource session, PersistenceContext persistenceContext) throws HibernateException {
146 
147 		LOG.debug( "Processing flush-time cascades" );
148 
149 		final Object anything = getAnything();
150 		//safe from concurrent modification because of how concurrentEntries() is implemented on IdentityMap
151 		for ( Map.Entry<Object,EntityEntry> me : persistenceContext.reentrantSafeEntityEntries() ) {
152 //		for ( Map.Entry me : IdentityMap.concurrentEntries( persistenceContext.getEntityEntries() ) ) {
153 			EntityEntry entry = (EntityEntry) me.getValue();
154 			Status status = entry.getStatus();
155 			if ( status == Status.MANAGED || status == Status.SAVING || status == Status.READ_ONLY ) {
156 				cascadeOnFlush( session, entry.getPersister(), me.getKey(), anything );
157 			}
158 		}
159 	}
160 
161 	private void cascadeOnFlush(EventSource session, EntityPersister persister, Object object, Object anything)
162 	throws HibernateException {
163 		session.getPersistenceContext().incrementCascadeLevel();
164 		try {
165 			new Cascade( getCascadingAction(), Cascade.BEFORE_FLUSH, session )
166 			.cascade( persister, object, anything );
167 		}
168 		finally {
169 			session.getPersistenceContext().decrementCascadeLevel();
170 		}
171 	}
172 
173 	protected Object getAnything() { return null; }
174 
175 	protected CascadingAction getCascadingAction() {
176 		return CascadingAction.SAVE_UPDATE;
177 	}

179 	/**
180 	 * Initialize the flags of the CollectionEntry, including the
181 	 * dirty check.
182 	 */
183 	private void prepareCollectionFlushes(PersistenceContext persistenceContext) throws HibernateException {
184 
185 		// Initialize dirty flags for arrays + collections with composite elements
186 		// and reset reached, doupdate, etc.
187 
188 		LOG.debug( "Dirty checking collections" );
189 
190 		for ( Map.Entry<PersistentCollection,CollectionEntry> entry :
191 				IdentityMap.concurrentEntries( (Map<PersistentCollection,CollectionEntry>) persistenceContext.getCollectionEntries() )) {
192 			entry.getValue().preFlush( entry.getKey() );
193 		}
194 	}
195 
196 	/**
197 	 * 1. detect any dirty entities
198 	 * 2. schedule any entity updates
199 	 * 3. search out any reachable collections
200 	 */
201 	private int flushEntities(final FlushEvent event, final PersistenceContext persistenceContext) throws HibernateException {
202 
203 		LOG.trace( "Flushing entities and processing referenced collections" );
204 
205 		final EventSource source = event.getSession();
206 		final Iterable<FlushEntityEventListener> flushListeners = source.getFactory().getServiceRegistry()
207 				.getService( EventListenerRegistry.class )
208 				.getEventListenerGroup( EventType.FLUSH_ENTITY )
209 				.listeners();
210 
211 		// Among other things, updateReachables() will recursively load all
212 		// collections that are moving roles. This might cause entities to
213 		// be loaded.
214 
215 		// So this needs to be safe from concurrent modification problems.
216 
217 		final Map.Entry<Object,EntityEntry>[] entityEntries = persistenceContext.reentrantSafeEntityEntries();
218 		final int count = entityEntries.length;
219 
220 		for ( Map.Entry<Object,EntityEntry> me : entityEntries ) {
221 
222 			// Update the status of the object and if necessary, schedule an update
223 
224 			EntityEntry entry = me.getValue();
225 			Status status = entry.getStatus();
226 
227 			if ( status != Status.LOADING && status != Status.GONE ) {
228 				final FlushEntityEvent entityEvent = new FlushEntityEvent( source, me.getKey(), entry );
229 				for ( FlushEntityEventListener listener : flushListeners ) {
230 					listener.onFlushEntity( entityEvent );
231 				}
232 			}
233 		}
234 
235 		source.getActionQueue().sortActions();
236 
237 		return count;
238 	}

240 	/**
241 	 * process any unreferenced collections and then inspect all known collections,
242 	 * scheduling creates/removes/updates
243 	 */
244 	@SuppressWarnings("unchecked")
245 	private int flushCollections(final EventSource session, final PersistenceContext persistenceContext) throws HibernateException {
246 		LOG.trace( "Processing unreferenced collections" );
247 
248 		final Map.Entry<PersistentCollection,CollectionEntry>[] entries = IdentityMap.concurrentEntries(
249 				(Map<PersistentCollection,CollectionEntry>) persistenceContext.getCollectionEntries()
250 		);
251 
252 		final int count = entries.length;
253 
254 		for ( Map.Entry<PersistentCollection,CollectionEntry> me : entries ) {
255 			CollectionEntry ce = me.getValue();
256 			if ( !ce.isReached() && !ce.isIgnore() ) {
257 				Collections.processUnreachableCollection( me.getKey(), session );
258 			}
259 		}
260 
261 		// Schedule updates to collections:
262 
263 		LOG.trace( "Scheduling collection removes/(re)creates/updates" );
264 
265 		ActionQueue actionQueue = session.getActionQueue();
266 		for ( Map.Entry<PersistentCollection,CollectionEntry> me :
267 			IdentityMap.concurrentEntries( (Map<PersistentCollection,CollectionEntry>) persistenceContext.getCollectionEntries() )) {
268 			PersistentCollection coll = me.getKey();
269 			CollectionEntry ce = me.getValue();
270 
271 			if ( ce.isDorecreate() ) {
272 				session.getInterceptor().onCollectionRecreate( coll, ce.getCurrentKey() );
273 				actionQueue.addAction(
274 						new CollectionRecreateAction(
275 								coll,
276 								ce.getCurrentPersister(),
277 								ce.getCurrentKey(),
278 								session
279 							)
280 					);
281 			}
282 			if ( ce.isDoremove() ) {
283 				session.getInterceptor().onCollectionRemove( coll, ce.getLoadedKey() );
284 				actionQueue.addAction(
285 						new CollectionRemoveAction(
286 								coll,
287 								ce.getLoadedPersister(),
288 								ce.getLoadedKey(),
289 								ce.isSnapshotEmpty(coll),
290 								session
291 							)
292 					);
293 			}
294 			if ( ce.isDoupdate() ) {
295 				session.getInterceptor().onCollectionUpdate( coll, ce.getLoadedKey() );
296 				actionQueue.addAction(
297 						new CollectionUpdateAction(
298 								coll,
299 								ce.getLoadedPersister(),
300 								ce.getLoadedKey(),
301 								ce.isSnapshotEmpty(coll),
302 								session
303 							)
304 					);
305 			}
306 			if ( !coll.wasInitialized() && coll.hasQueuedOperations() ) {
307 				actionQueue.addAction(
308 						new QueuedOperationCollectionAction(
309 								coll,
310 								ce.getLoadedPersister(),
311 								ce.getLoadedKey(),
312 								session
313 							)
314 					);
315 			}
316 
317 		}
318 
319 		actionQueue.sortCollectionActions();
320 
321 		return count;
322 	}

324 	/**
325 	 * Execute all SQL (and second-level cache updates) in a special order so that foreign-key constraints cannot
326 	 * be violated: <ol>
327 	 * <li> Inserts, in the order they were performed
328 	 * <li> Updates
329 	 * <li> Deletion of collection elements
330 	 * <li> Insertion of collection elements
331 	 * <li> Deletes, in the order they were performed
332 	 * </ol>
333 	 *
334 	 * @param session The session being flushed
335 	 */
336 	protected void performExecutions(EventSource session) {
337 		LOG.trace( "Executing flush" );
338 
339 		// IMPL NOTE : here we alter the flushing flag of the persistence context to allow
340 		//		during-flush callbacks more leniency in regards to initializing proxies and
341 		//		lazy collections during their processing.
342 		// For more information, see HHH-2763
343 		try {
344 			session.getTransactionCoordinator().getJdbcCoordinator().flushBeginning();
345 			session.getPersistenceContext().setFlushing( true );
346 			// we need to lock the collection caches before executing entity inserts/updates in order to
347 			// account for bi-directional associations
348 			session.getActionQueue().prepareActions();
349 			session.getActionQueue().executeActions();
350 		}
351 		finally {
352 			session.getPersistenceContext().setFlushing( false );
353 			session.getTransactionCoordinator().getJdbcCoordinator().flushEnding();
354 		}
355 	}

362 	/**
363 	 * 1. Recreate the collection key -> collection map
364 	 * 2. rebuild the collection entries
365 	 * 3. call Interceptor.postFlush()
366 	 */
367 	protected void postFlush(SessionImplementor session) throws HibernateException {
368 
369 		LOG.trace( "Post flush" );
370 
371 		final PersistenceContext persistenceContext = session.getPersistenceContext();
372 		persistenceContext.getCollectionsByKey().clear();
373 		
374 		// the database has changed now, so the subselect results need to be invalidated
375 		// the batch fetching queues should also be cleared - especially the collection batch fetching one
376 		persistenceContext.getBatchFetchQueue().clear();
377 
378 		for ( Map.Entry<PersistentCollection, CollectionEntry> me : IdentityMap.concurrentEntries( persistenceContext.getCollectionEntries() ) ) {
379 			CollectionEntry collectionEntry = me.getValue();
380 			PersistentCollection persistentCollection = me.getKey();
381 			collectionEntry.postFlush(persistentCollection);
382 			if ( collectionEntry.getLoadedPersister() == null ) {
383 				//if the collection is dereferenced, remove from the session cache
384 				//iter.remove(); //does not work, since the entrySet is not backed by the set
385 				persistenceContext.getCollectionEntries()
386 						.remove(persistentCollection);
387 			}
388 			else {
389 				//otherwise recreate the mapping between the collection and its key
390 				CollectionKey collectionKey = new CollectionKey(
391 						collectionEntry.getLoadedPersister(),
392 						collectionEntry.getLoadedKey()
393 				);
394 				persistenceContext.getCollectionsByKey().put(collectionKey, persistentCollection);
395 			}
396 		}
397 
398 	}
399 
400 	protected void postPostFlush(SessionImplementor session) {
401 		session.getInterceptor().postFlush( new LazyIterator( session.getPersistenceContext().getEntitiesByKey() ) );
402 	}
