45  /**
46   * A SQL dialect for TimesTen 5.1.
47   * 
48   * Known limitations:
49   * joined-subclass support because of no CASE support in TimesTen
50   * No support for subqueries that includes aggregation
51   *  - size() in HQL not supported
52   *  - user queries that does subqueries with aggregation
53   * No CLOB/BLOB support 
54   * No cascade delete support.
55   * No Calendar support
56   * No support for updating primary keys.
57   * 
58   * @author Sherry Listgarten and Max Andersen
59   */
60  public class TimesTenDialect extends Dialect {
61  	
62  	public TimesTenDialect() {
63  		super();
64  		registerColumnType( Types.BIT, "TINYINT" );
65  		registerColumnType( Types.BIGINT, "BIGINT" );
66  		registerColumnType( Types.SMALLINT, "SMALLINT" );
67  		registerColumnType( Types.TINYINT, "TINYINT" );
68  		registerColumnType( Types.INTEGER, "INTEGER" );
69  		registerColumnType( Types.CHAR, "CHAR(1)" );
70  		registerColumnType( Types.VARCHAR, "VARCHAR($l)" );
71  		registerColumnType( Types.FLOAT, "FLOAT" );
72  		registerColumnType( Types.DOUBLE, "DOUBLE" );
73  		registerColumnType( Types.DATE, "DATE" );
74  		registerColumnType( Types.TIME, "TIME" );
75  		registerColumnType( Types.TIMESTAMP, "TIMESTAMP" );
76  		registerColumnType( Types.VARBINARY, "VARBINARY($l)" );
77  		registerColumnType( Types.NUMERIC, "DECIMAL($p, $s)" );
78  		// TimesTen has no BLOB/CLOB support, but these types may be suitable 
79  		// for some applications. The length is limited to 4 million bytes.
80          registerColumnType( Types.BLOB, "VARBINARY(4000000)" ); 
81          registerColumnType( Types.CLOB, "VARCHAR(4000000)" );
82  	
83  		getDefaultProperties().setProperty(Environment.USE_STREAMS_FOR_BINARY, "true");
84  		getDefaultProperties().setProperty(Environment.STATEMENT_BATCH_SIZE, DEFAULT_BATCH_SIZE);
85  		registerFunction( "lower", new StandardSQLFunction("lower") );
86  		registerFunction( "upper", new StandardSQLFunction("upper") );
87  		registerFunction( "rtrim", new StandardSQLFunction("rtrim") );
88  		registerFunction( "concat", new StandardSQLFunction("concat", StandardBasicTypes.STRING) );
89  		registerFunction( "mod", new StandardSQLFunction("mod") );
90  		registerFunction( "to_char", new StandardSQLFunction("to_char",StandardBasicTypes.STRING) );
91  		registerFunction( "to_date", new StandardSQLFunction("to_date",StandardBasicTypes.TIMESTAMP) );
92  		registerFunction( "sysdate", new NoArgSQLFunction("sysdate", StandardBasicTypes.TIMESTAMP, false) );
93  		registerFunction( "getdate", new NoArgSQLFunction("getdate", StandardBasicTypes.TIMESTAMP, false) );
94  		registerFunction( "nvl", new StandardSQLFunction("nvl") );
95  
96  	}

175 	public String getLimitString(String querySelect, int offset, int limit) {
176 		if ( offset > 0 ) {
177 			throw new UnsupportedOperationException( "query result offset is not supported" );
178 		}
179 		return new StringBuilder( querySelect.length() + 8 )
180 				.append( querySelect )
181 				.insert( 6, " first " + limit )
182 				.toString();
183 	}

214 	public LockingStrategy getLockingStrategy(Lockable lockable, LockMode lockMode) {
215 		// TimesTen has no known variation of a "SELECT ... FOR UPDATE" syntax...
216 		if ( lockMode==LockMode.PESSIMISTIC_FORCE_INCREMENT) {
217 			return new PessimisticForceIncrementLockingStrategy( lockable, lockMode);
218 		}
219 		else if ( lockMode==LockMode.PESSIMISTIC_WRITE) {
220 			return new PessimisticWriteUpdateLockingStrategy( lockable, lockMode);
221 		}
222 		else if ( lockMode==LockMode.PESSIMISTIC_READ) {
223 			return new PessimisticReadUpdateLockingStrategy( lockable, lockMode);
224 		}
225 		else if ( lockMode==LockMode.OPTIMISTIC) {
226 			return new OptimisticLockingStrategy( lockable, lockMode);
227 		}
228 		else if ( lockMode==LockMode.OPTIMISTIC_FORCE_INCREMENT) {
229 			return new OptimisticForceIncrementLockingStrategy( lockable, lockMode);
230 		}
231 		else if ( lockMode.greaterThan( LockMode.READ ) ) {
232 			return new UpdateLockingStrategy( lockable, lockMode );
233 		}
234 		else {
235 			return new SelectLockingStrategy( lockable, lockMode );
236 		}
237 	}
