268 	private static boolean isNull(int[] status) {
269 		return status == null || status.length == 0;
270 	}
271 
272     private int appendWhere(StringBuffer jbossQl, Object[] args, String attrName,
273             int[] status, Timestamp after, Timestamp before) {
274         if (!isNull(status)) {
275             jbossQl.append(" WHERE m.mediaStatus IN (").append(status[0]);
276             for (int i = 1; i < status.length; i++) {
277 				jbossQl.append(", ").append(status[i]);
278 			}            
279 			jbossQl.append(")");
280         }
281         int i = 0;
282         if (after != null || before != null) {
283             jbossQl.append(isNull(status) ? " WHERE " : " AND ");
284             jbossQl.append(attrName);
285             if (after != null) {
286                 args[i++] = after;
287                 if (before == null) {
288                     jbossQl.append(" > ?").append(i);
289                 } else {
290                     jbossQl.append(" BETWEEN ?").append(i);
291                     args[i++] = before;
292                     jbossQl.append(" AND ?").append(i);
293                 }
294             } else if (before != null) {
295                 args[i++] = before;
296                 jbossQl.append(" < ?").append(i);
297             }
298         }
299         return i;
300     }
301     
302     private Collection findBy(String attrName, int[] status, Timestamp after,
303             Timestamp before, Integer offset, Integer limit,
304             boolean desc) throws FinderException {
305         // generate JBossQL query
306         int argsCount = 0;
307         if (after != null)
308             ++argsCount;
309         if (before != null)
310             ++argsCount;
311         if (offset != null)
312             ++argsCount;
313         if (limit != null)
314             ++argsCount;
315         Object[] args = new Object[argsCount];
316         StringBuffer jbossQl = new StringBuffer("SELECT OBJECT(m) FROM Media m");
317         int i = appendWhere(jbossQl, args, attrName, status, after, before);
318         jbossQl.append(" ORDER BY ");
319         jbossQl.append(attrName);
320         jbossQl.append(desc ? " DESC" : " ASC");
321         if (offset != null) {
322             args[i++] = offset;
323             jbossQl.append(" OFFSET ?").append(i);
324         }
325         if (limit != null) {
326             args[i++] = limit;
327             jbossQl.append(" LIMIT ?").append(i);
328         }
329         if (log.isDebugEnabled())
330             log.debug("Execute JBossQL: " + jbossQl);
331         // call dynamic-ql query
332         return ejbSelectGeneric(jbossQl.toString(), args);
333     }
334 
335     private int countBy(String attrName, int[] status, Timestamp after,
336             Timestamp before) throws FinderException {
337         // generate JBossQL query
338         int argsCount = 0;
339         if (after != null)
340             ++argsCount;
341         if (before != null)
342             ++argsCount;
343         Object[] args = new Object[argsCount];
344         StringBuffer jbossQl = new StringBuffer("SELECT COUNT(m) FROM Media m");
345         appendWhere(jbossQl, args, attrName, status, after, before);
346         // call dynamic-ql query
347         return ejbSelectGenericInt(jbossQl.toString(), args);
348     }
