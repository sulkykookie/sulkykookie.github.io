1   package org.infinispan.query;
2   
3   /**
4    * @author <a href="mailto:mluksa@redhat.com">Marko Luksa</a>
5    */
6   public class FetchOptions {
7   
8      private FetchMode fetchMode;
9   
10     private int fetchSize;
11  
12     public FetchOptions() {
13        fetchMode(FetchMode.LAZY);
14        fetchSize(1);
15     }
16  
17     /**
18      * Set the fetch mode to be used to fetch matching results
19      * @param fetchMode
20      * @return {@code this} to allow method chaining
21      */
22     public FetchOptions fetchMode(FetchMode fetchMode) {
23        if (fetchMode == null) {
24           throw new IllegalArgumentException("fetchMode should not be null");
25        }
26        this.fetchMode = fetchMode;
27        return this;
28     }
29  
30     /**
31      * Set the fetch size for batch loading of matches
32      * @param fetchSize
33      * @return {@code this} to allow method chaining
34      */
35     public FetchOptions fetchSize(int fetchSize) {
36        if (fetchSize < 1) {
37           throw new IllegalArgumentException("fetchSize should be greater than 0");
38        }
39        this.fetchSize = fetchSize;
40        return this;
41     }
42  
43     /**
44      * @return the selected fetch mode
45      */
46     public FetchMode getFetchMode() {
47        return fetchMode;
48     }
49  
50     /**
51      * @return the used fetch size
52      */
53     public int getFetchSize() {
54        return fetchSize;
55     }
56  
57     /**
58      * Specifies the fetching strategy
59      * for query results.
60      */
61      public static enum FetchMode {
62  
63          /**
64           * With eager mode all results are loaded as
65           * soon as the query is performed; this results
66           * in a larger initial transfer of entries but no
67           * remote operations during iteration of the resultset.
68           */
69          EAGER,
70  
71          /**
72           * With lazy loading the entries are not loaded
73           * until each one is specifically requested.
74           * If iterating on very larger result sets this
75           * is recommended to avoid loading too many entries
76           * in the VM performing the iteration.
77           */
78          LAZY
79      }
80  }
